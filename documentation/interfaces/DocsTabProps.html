import React, {useState, useCallback, useEffect} from 'react';
import {
  View,
  Text,
  FlatList,
  ActivityIndicator,
  RefreshControl,
  Pressable,
} from 'react-native';
import {StyleSheet} from 'react-native';
import {Button, ProgressBar, Chip} from 'react-native-paper';
import MaterialIcons from 'react-native-vector-icons/MaterialIcons';
import {SafeAreaContainer} from '../../../components';
import {DocumentPickerModal} from '../../../components/forms';
import {useKycDocuments} from '../../../hooks/useKycDocuments';
import type {KycDocView} from '../../../hooks/useKycDocuments';
import {useKycUpload} from '../../../hooks/useKycUpload';
import {useRefreshControl} from '../../../hooks/useRefreshControl';
import {useAppDispatch} from '../../../hooks/useTypedRedux';
import {setError as setUIError} from '../../../store/uiSlice';
import {spacing} from '../../../theme/palette';
import {useAppTheme} from '../../../theme/ThemeProvider';
import type {RNFile} from '../../../types/File';
import type {
  KycDocumentStatus,
  KycDocumentType,
} from '../../../types/KycDocument';

interface DocsTabProps {
  /** Lead ID to fetch documents for */
  leadId: string;
}

/**
 * Get status badge color configuration
 */
const getStatusConfig = (status: KycDocumentStatus | 'Missing') => {
  const configs = {
    Approved: {backgroundColor: '#4CAF50', textColor: '#FFFFFF'},
    Pending: {backgroundColor: '#FF9800', textColor: '#FFFFFF'},
    Rejected: {backgroundColor: '#F44336', textColor: '#FFFFFF'},
    Missing: {backgroundColor: '#9E9E9E', textColor: '#FFFFFF'},
  };
  return configs[status] || configs.Missing;
};

/**
 * Get document type icon
 */
const getDocumentIcon = (docType: KycDocumentType): string => {
  const iconMap = {
    IncomeProof: 'account-balance-wallet',
    LightBill: 'electrical-services',
    CancelledCheque: 'account-balance',
    Aadhaar: 'badge',
    PAN: 'credit-card',
    PassportPhoto: 'person',
    Other: 'insert-drive-file',
  };
  return iconMap[docType] || 'insert-drive-file';
};

/**
 * Document List Item Component
 */
const DocumentListItem: React.FC<{
  document: KycDocView;
  leadId: string;
  onUploadSuccess: () => void;
  onUploadError: (error: Error) => void;
}> = ({document, leadId, onUploadSuccess, onUploadError}) => {
  const theme = useAppTheme();
  const [showPicker, setShowPicker] = useState(false);

  // Upload hook
  const {
    startUpload,
    progress,
    status: uploadStatus,
    error,
    cancel,
  } = useKycUpload();
  const statusConfig = getStatusConfig(document.status);
  const documentIcon = getDocumentIcon(document.docType);

  /**
   * Handle file selection from picker
   */
  const handleFilePicked = useCallback(
    (file: RNFile) => {
      console.log(`[DocsTab] File picked for ${document.docType}:`, file);
      // For lead-specific uploads, we need to pass both customerId and leadId
      // The useKycUpload hook handles customerId, but we need leadId context
      startUpload(document.docType, file, leadId);
    },
    [document.docType, startUpload, leadId]
  );

  /**
   * Handle upload button press
   */
  const handleUploadPress = useCallback(() => {
    if (!document.canUpload) {
      console.warn(`[DocsTab] Upload not allowed for ${document.docType}`);
      return;
    }
    setShowPicker(true);
  }, [document.canUpload, document.docType]);

  /**
   * Handle upload success
   */
  useEffect(() => {
    if (uploadStatus === 'success') {
      console.log(`[DocsTab] Upload successful for ${document.docType}`);
      onUploadSuccess();
    }
  }, [uploadStatus, document.docType, onUploadSuccess]);

  /**
   * Handle upload error
   */
  useEffect(() => {
    if (uploadStatus === 'error' && error) {
      console.error(`[DocsTab] Upload failed for ${document.docType}:`, error);
      onUploadError(error);
    }
  }, [uploadStatus, error, document.docType, onUploadError]);

  const isUploading = uploadStatus === 'running';
  const showUploadButton = document.canUpload && !isUploading;

  return (
    <Pressable
      style={[styles.documentItem, {backgroundColor: theme.colors.surface}]}
      testID={`doc-item-${document.docType}`}
    >
      {/* Document Header */}
      <View style={styles.documentHeader}>
        {/* Document Icon */}
        <MaterialIcons
          name={documentIcon}
          size={24}
          color={theme.colors.primary}
          style={styles.documentIcon}
        />

        {/* Document Content */}
        <View style={styles.documentContent}>
          <Text
            style={[styles.documentName, {color: theme.colors.onSurface}]}
            testID={`doc-name-${document.docType}`}
          >
            {document.displayName}
          </Text>

          {/* Upload Date */}
          {document.latestDoc && (
            <Text
              style={[
                styles.uploadDate,
                {color: theme.colors.onSurfaceVariant},
              ]}
              testID={`doc-date-${document.docType}`}
            >
              Uploaded:{' '}
              {new Date(document.latestDoc.uploadedAt).toLocaleDateString()}
            </Text>
          )}
        </View>

        {/* Status Badge */}
        <Chip
          mode="flat"
          compact={true}
          style={[
            styles.statusChip,
            {backgroundColor: statusConfig.backgroundColor},
          ]}
          textStyle={[styles.statusText, {color: statusConfig.textColor}]}
          testID={`doc-status-${document.docType}`}
        >
          {document.status}
        </Chip>
      </View>

      {/* Action Section */}
      <View style={styles.actionSection}>
        {/* Progress Bar (during upload) */}
        {isUploading && progress !== null && (
          <View style={styles.progressSection}>
            <ProgressBar
              progress={progress / 100}
              color={theme.colors.primary}
              style={styles.progressBar}
              testID={`doc-progress-${document.docType}`}
            />
            <View style={styles.progressActions}>
              <Text
                style={[
                  styles.progressText,
                  {color: theme.colors.onSurfaceVariant},
                ]}
              >
                {progress}%
              </Text>
              <Button
                mode="text"
                onPress={cancel}
                compact
                textColor={theme.colors.error}
                testID={`doc-cancel-${document.docType}`}
              >
                Cancel
              </Button>
            </View>
          </View>
        )}

        {/* Upload Button */}
        {showUploadButton && (
          <Button
            mode="outlined"
            onPress={handleUploadPress}
            icon={({size, color}) => (
              <MaterialIcons name="cloud-upload" size={size} color={color} />
            )}
            style={styles.uploadButton}
            contentStyle={styles.uploadButtonContent}
            labelStyle={styles.uploadButtonLabel}
            testID={`doc-upload-${document.docType}`}
          >
            Upload
          </Button>
        )}

        {/* Status Message for Non-Uploadable Documents */}
        {!document.canUpload && !isUploading && (
          <Text
            style={[
              styles.statusMessage,
              {color: theme.colors.onSurfaceVariant},
            ]}
            testID={`doc-message-${document.docType}`}
          >
            {document.status === 'Approved'
              ? 'Document approved'
              : 'Upload not available'}
          </Text>
        )}
      </View>

      {/* Document Picker Modal */}
      <DocumentPickerModal
        visible={showPicker}
        onClose={() => setShowPicker(false)}
        onFilePicked={handleFilePicked}
        testID={`doc-picker-${document.docType}`}
      />
    </Pressable>
  );
};

/**
 * DocsTab Component
 * Shows KYC documents for a specific lead with upload functionality
 */
const DocsTab: React.FC<DocsTabProps> = ({leadId}) => {
  const theme = useAppTheme();
  const dispatch = useAppDispatch();

  // Fetch lead-specific KYC documents
  const {docs, isLoading, refetch} = useKycDocuments(undefined, leadId);

  // Enhanced refresh control
  const {refreshKyc, canRefreshKyc} = useRefreshControl();

  /**
   * Handle pull-to-refresh
   */
  const handleRefresh = useCallback(async () => {
    console.log('[DocsTab] Refreshing KYC documents for lead:', leadId);
    try {
      await refreshKyc();
      refetch(); // Also trigger local refetch
    } catch (error) {
      console.error('[DocsTab] Refresh error:', error);
    }
  }, [refreshKyc, refetch, leadId]);

  /**
   * Handle upload success - refresh data
   */
  const handleUploadSuccess = useCallback(() => {
    console.log('[DocsTab] Upload successful, refreshing data...');
    refetch(); // Refresh local data
  }, [refetch]);

  /**
   * Handle upload error - show global error
   */
  const handleUploadError = useCallback(
    (error: Error) => {
      console.error('[DocsTab] Upload error:', error);
      dispatch(setUIError(error.message));
    },
    [dispatch]
  );

  /**
   * Render individual document item
   */
  const renderDocumentItem = useCallback(
    ({item}: {item: KycDocView}) => (
      <DocumentListItem
        document={item}
        leadId={leadId}
        onUploadSuccess={handleUploadSuccess}
        onUploadError={handleUploadError}
      />
    ),
    [leadId, handleUploadSuccess, handleUploadError]
  );

  /**
   * Get key extractor for FlatList
   */
  const keyExtractor = useCallback((item: KycDocView) => item.docType, []);

  // Loading state
  if (isLoading && !docs) {
    return (
      <SafeAreaContainer>
        <View
          style={[
            styles.loadingContainer,
            {backgroundColor: theme.colors.background},
          ]}
        >
          <ActivityIndicator size="large" color={theme.colors.primary} />
          <Text
            style={[styles.loadingText, {color: theme.colors.onSurfaceVariant}]}
          >
            Loading documents...
          </Text>
        </View>
      </SafeAreaContainer>
    );
  }

  // Content state
  return (
    <SafeAreaContainer>
      <View
        style={[styles.container, {backgroundColor: theme.colors.background}]}
        testID="docs-tab-container"
      >
        <FlatList
          data={docs}
          renderItem={renderDocumentItem}
          keyExtractor={keyExtractor}
          ListEmptyComponent={<EmptyDocsState />}
          contentContainerStyle={styles.listContent}
          showsVerticalScrollIndicator={false}
          refreshControl={
            <RefreshControl
              refreshing={isLoading && canRefreshKyc()}
              onRefresh={handleRefresh}
              colors={[theme.colors.primary]}
              tintColor={theme.colors.primary}
            />
          }
          // Performance optimizations
          removeClippedSubviews={true}
          maxToRenderPerBatch={7} // We have 7 KYC document types
          windowSize={10}
          initialNumToRender={7}
          getItemLayout={(data, index) => ({
            length: 120, // Approximate item height
            offset: 120 * index,
            index,
          })}
          testID="docs-list"
        />
      </View>
    </SafeAreaContainer>
  );
};
