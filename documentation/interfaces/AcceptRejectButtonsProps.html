import React from 'react';
import {View, StyleSheet} from 'react-native';
import {Button} from '../common';

/**
 * Props interface for AcceptRejectButtons component
 */
export interface AcceptRejectButtonsProps {
  /** Callback fired when Accept button is pressed */
  onAccept: () => void;
  /** Callback fired when Reject button is pressed */
  onReject: () => void;
  /** Whether both buttons should be disabled */
  disabled?: boolean;
  /** Whether buttons should show loading state */
  loading?: boolean;
  /** Custom label for Accept button (default: "Accept") */
  acceptLabel?: string;
  /** Custom label for Reject button (default: "Reject") */
  rejectLabel?: string;
  /** Test ID for the component container */
  testID?: string;
}

/**
 * AcceptRejectButtons Component
 *
 * Renders Accept and Reject buttons with proper theming, accessibility,
 * and WCAG 2.1 AA compliance. Accept button uses green primary styling,
 * Reject button uses red secondary styling.
 *
 * @param props - Component props
 * @returns JSX element containing both action buttons
 */
const AcceptRejectButtons: React.FC<AcceptRejectButtonsProps> = ({
  onAccept,
  onReject,
  disabled = false,
  loading = false,
  acceptLabel = 'Accept',
  rejectLabel = 'Reject',
  testID = 'accept-reject-buttons',
}) => {
  return (
    <View style={styles.container} testID={testID}>
      <Button
        title={acceptLabel}
        onPress={onAccept}
        variant="primary"
        size="medium"
        disabled={disabled}
        loading={loading}
        style={styles.acceptButton}
        testID={`${testID}-accept`}
        accessibilityLabel={`${acceptLabel} quotation`}
        accessibilityHint="Double tap to accept this quotation"
        accessibilityRole="button"
      />
      <Button
        title={rejectLabel}
        onPress={onReject}
        variant="outline"
        size="medium"
        disabled={disabled}
        loading={loading}
        style={styles.rejectButton}
        textStyle={styles.rejectButtonText}
        testID={`${testID}-reject`}
        accessibilityLabel={`${rejectLabel} quotation`}
        accessibilityHint="Double tap to reject this quotation"
        accessibilityRole="button"
      />
    </View>
  );
};

/**
 * Styles for AcceptRejectButtons component
 * Ensures proper spacing, colors, and minimum touch targets (44px)
 */
const styles = StyleSheet.create({
  container: {
    flexDirection: 'row',
    gap: 12,
    paddingVertical: 8,
    alignItems: 'center',
  },
  acceptButton: {
    flex: 1,
    backgroundColor: '#4CAF50', // Material Green 500 - WCAG AA compliant
    minHeight: 44, // WCAG minimum touch target
    borderRadius: 8,
  },
  rejectButton: {
    flex: 1,
    borderColor: '#F44336', // Material Red 500 - WCAG AA compliant
    borderWidth: 1,
    backgroundColor: 'transparent',
    minHeight: 44, // WCAG minimum touch target
    borderRadius: 8,
  },
  rejectButtonText: {
    color: '#F44336', // Material Red 500 - matches border
  },
});

export default AcceptRejectButtons;
