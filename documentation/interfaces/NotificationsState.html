export interface Notification {
  /** Unique identifier for the notification (e.g., "NOTIF-001") */
  id: string;
  /** Display title of the notification */
  title: string;
  /** Main notification message content */
  message: string;
  /** Category/type of notification (e.g., 'user', 'announcement', 'system') */
  type: NotificationType;
  /** Severity level (e.g., 'info', 'warning', 'error') */
  severity: NotificationSeverity;
  /** Server-side read status (reference only - we manage read state locally) */
  read: boolean;
  /** ISO timestamp when notification was created on server */
  createdAt: string;
  /** Metadata for notification-specific data */
  metadata: Record<string, any>;
}

/**
 * API response structure for notifications endpoint
 * GET /api/v1/notifications response format
 */
export interface NotificationsApiResponse {
  success: boolean;
  data: {
    notifications: Notification[];
    total: number;
    unreadCount: number;
    page: number;
    limit: number;
    totalPages: number;
  };
}

/**
 * Available notification statuses for filtering
 */
export type NotificationStatus = 'read' | 'unread' | 'all';

/**
 * Available notification types for filtering
 */
export type NotificationType =
  | 'system'
  | 'security'
  | 'lead'
  | 'quotation'
  | 'commission'
  | 'user'
  | 'announcement';

/**
 * Available notification severities for filtering
 */
export type NotificationSeverity = 'info' | 'warning' | 'error';

/**
 * Available sort fields for notifications
 */
export type NotificationSortBy = 'createdAt' | 'readAt' | 'title';

/**
 * Available sort orders
 */
export type NotificationSortOrder = 'asc' | 'desc';

/**
 * Query parameters for notifications API
 * Used with GET /api/v1/notifications
 */
export interface NotificationsQueryParams {
  /** Page number for pagination (default: 1) */
  page?: number;
  /** Number of items per page (default: 20) */
  limit?: number;
  /** Filter by read status (default: all) */
  status?: NotificationStatus;
  /** Filter by notification type */
  type?: NotificationType;
  /** Filter notifications from this date (ISO string format) */
  dateFrom?: string;
  /** Filter notifications to this date (ISO string format) */
  dateTo?: string;
  /** Field to sort by (default: createdAt) */
  sortBy?: NotificationSortBy;
  /** Sort order (default: desc) */
  sortOrder?: NotificationSortOrder;
}

/**
 * Redux state shape for notifications slice
 * Manages local read state for immediate UI updates
 */
export interface NotificationsState {
  /** Array of all notifications from server */
  items: Notification[];
  /** Array of notification IDs marked as read locally (stored as array for Redux serialization) */
  readIds: string[];
  /** Calculated count of unread notifications based on local read state */
  unreadCount: number;
  /** Timestamp of last successful fetch from server */
  lastFetched: number | null;
  /** Flag indicating if polling is currently active */
  isPolling: boolean;
  /** Current error message from polling, if any */
  pollError: string | null;
}
