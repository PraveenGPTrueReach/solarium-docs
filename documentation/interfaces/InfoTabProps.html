import React from 'react';
import {
  View,
  Text,
  ScrollView,
  ActivityIndicator,
  RefreshControl,
} from 'react-native';
import MaterialIcons from 'react-native-vector-icons/MaterialIcons';
import {useGetLeadByIdQuery} from '../../../api/endpoints/leads';
import {useGetServiceByIdQuery} from '../../../api/endpoints/services';
import {SafeAreaContainer} from '../../../components';
import {useRefreshControl} from '../../../hooks/useRefreshControl';
import {useAppTheme} from '../../../theme/ThemeProvider';
import {formatDisplayDate} from '../../../utils/dateHelpers';
import styles from './InfoTab.styles';

interface InfoTabProps {
  /** Lead ID to fetch and display */
  leadId: string;
}

/**
 * Service Name Component
 * Fetches and displays individual service name
 */
const ServiceName: React.FC<{serviceId: string}> = ({serviceId}) => {
  const {data: service, isLoading} = useGetServiceByIdQuery(serviceId);

  if (isLoading) {
    return <Text>Loading...</Text>;
  }

  return <Text>{service?.name || serviceId}</Text>;
};

/**
 * Services List Component
 * Renders comma-separated list of service names
 */
const ServicesList: React.FC<{serviceIds: string[]}> = ({serviceIds}) => {
  const theme = useAppTheme();

  return (
    <View style={styles.servicesContainer}>
      {serviceIds.map((serviceId, index) => (
        <View key={serviceId} style={styles.serviceItem}>
          <ServiceName serviceId={serviceId} />
          {index < serviceIds.length - 1 && (
            <Text
              style={[styles.serviceSeparator, {color: theme.colors.onSurface}]}
            >
              ,{' '}
            </Text>
          )}
        </View>
      ))}
    </View>
  );
};

/**
 * Detail Row Component
 * Renders a single detail row with label and value
 */
const DetailRow: React.FC<{
  label: string;
  value: string | React.ReactNode;
  icon?: string;
}> = ({label, value, icon}) => {
  const theme = useAppTheme();

  return (
    <View style={styles.detailRow}>
      <View style={styles.detailLabel}>
        {icon && (
          <MaterialIcons
            name={icon}
            size={16}
            color={theme.colors.onSurfaceVariant}
            style={styles.detailIcon}
          />
        )}
        <Text
          style={[styles.labelText, {color: theme.colors.onSurfaceVariant}]}
        >
          {label}:
        </Text>
      </View>
      <View style={styles.detailValue}>
        {typeof value === 'string' ? (
          <Text style={[styles.valueText, {color: theme.colors.onSurface}]}>
            {value}
          </Text>
        ) : (
          value
        )}
      </View>
    </View>
  );
};

/**
 * InfoTab Component
 * Shows essential lead information in a clean, organized layout
 */
const InfoTab: React.FC<InfoTabProps> = ({leadId}) => {
  const theme = useAppTheme();

  // Fetch lead details
  const {
    data: lead,
    isLoading,
    isError,
    isFetching,
  } = useGetLeadByIdQuery(leadId, {
    selectFromResult: result => ({
      ...result,
      // Extract the lead data from the success response format
      data: result.data?.success ? result.data.data : result.data,
    }),
  });

  // Enhanced refresh control with 60s debounce for detail screen
  const {refreshSingleLead, canRefreshSingleLead} = useRefreshControl();

  /**
   * Handle pull-to-refresh
   */
  const handleRefresh = React.useCallback(async () => {
    console.log('[InfoTab] Refreshing lead data for:', leadId);
    try {
      // Use the new single lead refresh method
      await refreshSingleLead(leadId);
    } catch (error) {
      console.error('[InfoTab] Refresh error:', error);
    }
  }, [refreshSingleLead, leadId]);

  // Loading state
  if (isLoading && !lead) {
    return (
      <SafeAreaContainer>
        <View style={styles.loadingContainer}>
          <ActivityIndicator size="large" color={theme.colors.primary} />
          <Text
            style={[styles.loadingText, {color: theme.colors.onSurfaceVariant}]}
          >
            Loading Lead Info...
          </Text>
        </View>
      </SafeAreaContainer>
    );
  }

  // Error state
  if (isError || !lead) {
    return (
      <SafeAreaContainer>
        <View style={styles.errorContainer}>
          <MaterialIcons
            name="error-outline"
            size={64}
            color={theme.colors.error}
          />
          <Text style={[styles.errorTitle, {color: theme.colors.error}]}>
            Failed to Load Service Request
          </Text>
          <Text
            style={[
              styles.errorDescription,
              {color: theme.colors.onSurfaceVariant},
            ]}
          >
            Please pull down to refresh and try again.
          </Text>
        </View>
      </SafeAreaContainer>
    );
  }

  return (
    <SafeAreaContainer>
      <ScrollView
        style={[styles.container, {backgroundColor: theme.colors.background}]}
        contentContainerStyle={styles.contentContainer}
        refreshControl={
          <RefreshControl
            refreshing={isFetching && canRefreshSingleLead()}
            onRefresh={handleRefresh}
            colors={[theme.colors.primary]}
            tintColor={theme.colors.primary}
          />
        }
        showsVerticalScrollIndicator={false}
      >
        {/* Details Section */}
        <View
          style={[styles.detailsCard, {backgroundColor: theme.colors.surface}]}
        >
          <DetailRow
            label="Requested Services"
            value={<ServicesList serviceIds={lead.services} />}
            icon="build"
          />

          <DetailRow
            label="Requested On"
            value={formatDisplayDate(lead.createdAt)}
            icon="schedule"
          />

          <DetailRow
            label="Last Updated"
            value={formatDisplayDate(lead.updatedAt)}
            icon="update"
          />

          {lead.followUpDate && (
            <DetailRow
              label="Next Follow-up"
              value={formatDisplayDate(lead.followUpDate)}
              icon="event"
            />
          )}

          <DetailRow
            label="Installation Address"
            value={lead.address}
            icon="location-on"
          />

          {lead.assignedTo && (
            <DetailRow
              label="Assigned Partner"
              value={lead.assignedTo}
              icon="person"
            />
          )}
        </View>
      </ScrollView>
    </SafeAreaContainer>
  );
};

export default InfoTab;
