import {createApi} from '@reduxjs/toolkit/query/react';
import type {
  QuotationListItem,
  QuotationDetail,
  QuotationsListResponse,
  QuotationPdfResponse,
  GetQuotationsRequest,
} from '../../types/Quotation';
import {baseQuery, transformResponse, transformError} from '../baseQuery';

/**
 * Response type for accept/reject quotation actions
 */
export interface QuotationActionResponse {
  success: boolean;
  error?: string;
}

/**
 * Transform quotation errors to provide better error messages
 */
const transformQuotationError = (error: any) => {
  const baseError = transformError(error);

  // Add quotation-specific error handling
  if (error.status === 404) {
    return {
      ...baseError,
      message: 'Quotation not found or access denied',
    };
  }

  if (error.status === 403) {
    return {
      ...baseError,
      message: 'You do not have permission to access this quotation',
    };
  }

  if (error.status === 409) {
    return {
      ...baseError,
      message: 'Quotation already processed',
      code: 409,
    };
  }

  return baseError;
};

/**
 * Quotations API Definition
 */
export const quotationsApi = createApi({
  reducerPath: 'quotationsApi',
  baseQuery,
  tagTypes: [
    'Quotation',
    'QuotationsList',
    'QuotationPdf',
    'Lead',
    'LeadsList',
  ],
  endpoints: builder => ({
    /**
     * Get quotations with filters
     */
    getQuotations: builder.query<QuotationsListResponse, GetQuotationsRequest>({
      query: (params = {}) => {
        const {offset = 0, limit = 25, ...filters} = params;
        return {
          url: 'quotations',
          params: {offset, limit, ...filters},
        };
      },
      transformResponse: transformResponse<QuotationsListResponse>,
      transformErrorResponse: transformQuotationError,
      providesTags: result => [
        ...(result?.items.map(item => ({
          type: 'Quotation' as const,
          id: item.quotationId,
        })) || []),
        {type: 'QuotationsList', id: 'LIST'},
      ],
    }),

    /**
     * Get quotations by lead ID
     */
    getQuotationsByLeadId: builder.query<QuotationListItem[], string>({
      query: leadId => ({
        url: 'quotations',
        params: {leadId, limit: 100}, // Get all quotations for the lead
      }),
      transformResponse: (response: any): QuotationListItem[] => {
        const data = transformResponse<QuotationsListResponse>(response);
        return data.items;
      },
      transformErrorResponse: transformQuotationError,
      providesTags: (result, error, leadId) => [
        ...(result?.map(item => ({
          type: 'Quotation' as const,
          id: item.quotationId,
        })) || []),
        {type: 'QuotationsList', id: leadId},
      ],
    }),

    /**
     * Get quotation by ID
     */
    getQuotationById: builder.query<QuotationDetail, string>({
      query: quotationId => `quotations/${quotationId}`,
      transformResponse: transformResponse<QuotationDetail>,
      transformErrorResponse: transformQuotationError,
      providesTags: (result, error, quotationId) => [
        {type: 'Quotation', id: quotationId},
      ],
    }),

    /**
     * Get quotation PDF URL
     */
    getQuotationPdf: builder.mutation<QuotationPdfResponse, string>({
      query: quotationId => ({
        url: `quotations/${quotationId}/pdf`,
        method: 'GET',
      }),
      transformResponse: transformResponse<QuotationPdfResponse>,
      transformErrorResponse: transformQuotationError,
      invalidatesTags: (result, error, quotationId) => [
        {type: 'QuotationPdf', id: quotationId},
      ],
    }),

    /**
     * Accept a quotation
     *
     * @remarks
     * This mutation handles the quotation acceptance workflow:
     * - Sends PATCH request to accept endpoint
     * - Invalidates relevant caches (quotation and lead data)
     * - Triggers automatic refresh of dependent data
     *
     * Backend side effects:
     * - Updates quotation status to 'Accepted'
     * - Updates lead status to 'Customer Accepted'
     * - Triggers any configured business logic workflows
     *
     * @param quotationId - Unique identifier for the quotation to accept
     * @returns Promise resolving to success/error response
     */
    acceptQuotation: builder.mutation<QuotationActionResponse, string>({
      query: quotationId => ({
        url: `quotations/${quotationId}/accept`,
        method: 'PATCH',
        body: {},
      }),
      transformErrorResponse: transformQuotationError,
      invalidatesTags: (result, error, quotationId) => [
        {type: 'Quotation', id: quotationId},
        {type: 'QuotationsList', id: 'LIST'},
        // Also invalidate lead data as quotation acceptance affects lead status
        {type: 'Lead', id: 'LIST'},
        {type: 'LeadsList', id: 'LIST'},
      ],
    }),

    /**
     * Reject a quotation
     *
     * @remarks
     * This mutation handles the quotation rejection workflow:
     * - Sends PATCH request to reject endpoint
     * - Invalidates relevant caches (quotation and lead data)
     * - Triggers automatic refresh of dependent data
     *
     * Backend side effects:
     * - Updates quotation status to 'Rejected'
     * - May update lead status based on business rules
     * - Triggers any configured business logic workflows
     *
     * @param quotationId - Unique identifier for the quotation to reject
     * @returns Promise resolving to success/error response
     */
    rejectQuotation: builder.mutation<QuotationActionResponse, string>({
      query: quotationId => ({
        url: `quotations/${quotationId}/reject`,
        method: 'PATCH',
        body: {},
      }),
      transformErrorResponse: transformQuotationError,
      invalidatesTags: (result, error, quotationId) => [
        {type: 'Quotation', id: quotationId},
        {type: 'QuotationsList', id: 'LIST'},
        // Also invalidate lead data as quotation rejection may affect lead
        {type: 'Lead', id: 'LIST'},
        {type: 'LeadsList', id: 'LIST'},
      ],
    }),
  }),
});

// Export hooks
export const {
  useGetQuotationsQuery,
  useGetQuotationsByLeadIdQuery,
  useGetQuotationByIdQuery,
  useGetQuotationPdfMutation,
  useAcceptQuotationMutation,
  useRejectQuotationMutation,
} = quotationsApi;

export default quotationsApi;
