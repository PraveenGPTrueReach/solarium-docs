import {formatDistanceToNow} from 'date-fns';
import React from 'react';
import {View, Text, StyleSheet, TouchableOpacity} from 'react-native';
import {Card} from 'react-native-paper';
import MaterialIcons from 'react-native-vector-icons/MaterialIcons';
import {useAppTheme} from '../../theme/ThemeProvider';
import type {
  Notification,
  NotificationSeverity,
} from '../../types/Notification';

export interface NotificationListItemProps {
  notification: Notification & {isRead: boolean};
  onPress: (notification: Notification) => void;
  testID?: string;
}

/**
 * Get icon name based on notification type
 */
const getNotificationIcon = (type: string): string => {
  switch (type) {
    case 'lead':
      return 'assignment';
    case 'quotation':
      return 'receipt';
    case 'commission':
      return 'attach-money';
    case 'system':
      return 'settings';
    case 'security':
      return 'security';
    case 'announcement':
      return 'campaign';
    default:
      return 'notifications';
  }
};

/**
 * Get severity color based on notification severity
 */
const getSeverityColor = (
  severity: NotificationSeverity,
  theme: any
): string => {
  switch (severity) {
    case 'error':
      return theme.colors.error;
    case 'warning':
      return theme.colors.tertiary;
    case 'info':
    default:
      return theme.colors.primary;
  }
};

const NotificationListItem: React.FC<NotificationListItemProps> = ({
  notification,
  onPress,
  testID,
}) => {
  const theme = useAppTheme();

  const handlePress = () => {
    onPress(notification);
  };

  const timeAgo = formatDistanceToNow(new Date(notification.createdAt), {
    addSuffix: true,
  });

  const iconName = getNotificationIcon(notification.type);
  const severityColor = getSeverityColor(notification.severity, theme);

  return (
    <TouchableOpacity
      style={styles.container}
      onPress={handlePress}
      accessible={true}
      accessibilityRole="button"
      accessibilityLabel={`${notification.title}. ${notification.message}. ${
        notification.isRead ? 'Read' : 'Unread'
      }. ${timeAgo}`}
      accessibilityHint="Tap to view details"
      testID={testID}
    >
      <Card
        style={[
          styles.card,
          {
            backgroundColor: notification.isRead
              ? theme.colors.surface
              : theme.colors.secondaryContainer,
          },
        ]}
        elevation={1}
      >
        <View style={styles.content}>
          {/* Left: Icon with severity color */}
          <View style={styles.iconContainer}>
            <MaterialIcons name={iconName} size={24} color={severityColor} />
          </View>

          {/* Center: Content */}
          <View style={styles.textContainer}>
            <Text
              style={[
                styles.title,
                {
                  color: theme.colors.onSurface,
                  fontWeight: notification.isRead ? '400' : '600',
                },
              ]}
              numberOfLines={1}
              ellipsizeMode="tail"
            >
              {notification.title}
            </Text>
            <Text
              style={[
                styles.message,
                {
                  color: theme.colors.onSurfaceVariant,
                  fontWeight: notification.isRead ? '400' : '500',
                },
              ]}
              numberOfLines={2}
              ellipsizeMode="tail"
            >
              {notification.message}
            </Text>
            <Text
              style={[styles.timestamp, {color: theme.colors.onSurfaceVariant}]}
            >
              {timeAgo}
            </Text>
          </View>

          {/* Right: Unread indicator */}
          {!notification.isRead && (
            <View style={styles.unreadIndicatorContainer}>
              <View
                style={[
                  styles.unreadIndicator,
                  {backgroundColor: theme.colors.primary},
                ]}
              />
            </View>
          )}
        </View>
      </Card>
    </TouchableOpacity>
  );
};

const styles = StyleSheet.create({
  container: {
    marginHorizontal: 16,
    marginVertical: 4,
  },
  card: {
    borderRadius: 8,
  },
  content: {
    flexDirection: 'row',
    alignItems: 'flex-start',
    padding: 16,
  },
  iconContainer: {
    marginRight: 12,
    marginTop: 2,
  },
  textContainer: {
    flex: 1,
    marginRight: 8,
  },
  title: {
    fontSize: 16,
    lineHeight: 20,
    marginBottom: 4,
  },
  message: {
    fontSize: 14,
    lineHeight: 18,
    marginBottom: 8,
  },
  timestamp: {
    fontSize: 12,
    lineHeight: 16,
  },
  unreadIndicatorContainer: {
    justifyContent: 'center',
    alignItems: 'center',
    marginLeft: 8,
  },
  unreadIndicator: {
    width: 8,
    height: 8,
    borderRadius: 4,
  },
});

export default NotificationListItem;
