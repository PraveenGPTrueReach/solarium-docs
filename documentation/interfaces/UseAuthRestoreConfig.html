import {useEffect, useRef} from 'react';
import {restoreAuth} from '../store/newAuthSlice';
import {useAppDispatch} from './useTypedRedux';

/**
 * Configuration for auth restoration
 */
export interface UseAuthRestoreConfig {
  /**
   * Whether to enable auth restoration
   * Default: true
   */
  enabled?: boolean;

  /**
   * Delay before running restoration (in ms)
   * Allows rehydration to complete fully
   * Default: 500ms
   */
  delay?: number;

  /**
   * Callback when restoration completes
   */
  onComplete?: (result: any) => void;

  /**
   * Callback when restoration fails
   */
  onError?: (error: any) => void;
}

/**
 * useAuthRestore Hook
 *
 * @param config Configuration options
 * @returns Object with restoration status and manual trigger function
 */
export const useAuthRestore = (config: UseAuthRestoreConfig = {}) => {
  const {enabled = true, delay = 500, onComplete, onError} = config;

  const dispatch = useAppDispatch();
  const hasRunRef = useRef(false);
  const isRunningRef = useRef(false);

  /**
   * Perform auth restoration
   */
  const performRestore = async (): Promise<void> => {
    if (isRunningRef.current || hasRunRef.current) {
      console.log('[useAuthRestore] Restoration already run or in progress');
      return;
    }

    isRunningRef.current = true;

    try {
      console.log('[useAuthRestore] Starting auth restoration...');

      const result = await dispatch(restoreAuth()).unwrap();

      console.log('[useAuthRestore] Restoration completed:', result);
      hasRunRef.current = true;
      onComplete?.(result);
    } catch (error) {
      console.error('[useAuthRestore] Restoration failed:', error);
      onError?.(error);
    } finally {
      isRunningRef.current = false;
    }
  };

  /**
   * Effect to run restoration on mount
   */
  useEffect(() => {
    if (!enabled) {
      console.log('[useAuthRestore] Auth restoration disabled');
      return;
    }

    const timeoutId = setTimeout(() => {
      performRestore();
    }, delay);

    return () => {
      clearTimeout(timeoutId);
    };
  }, [enabled, delay, performRestore]);

  /**
   * Manual trigger function for testing or special cases
   */
  const triggerRestore = async (): Promise<void> => {
    hasRunRef.current = false; // Allow re-run
    await performRestore();
  };

  return {
    isRunning: isRunningRef.current,
    hasRun: hasRunRef.current,
    triggerRestore,
  };
};

export default useAuthRestore;
