import {useCallback, useEffect, useMemo, useRef} from 'react';
import {AppState, AppStateStatus} from 'react-native';
import {RefreshResult} from '../refresh/refreshTypes';
import {useRefreshControl} from './useRefreshControl';

/**
 * Polling configuration
 */
interface TicketPollingConfig {
  /** Base polling interval in seconds (default: 30) */
  baseIntervalSeconds?: number;
  /** Maximum jitter in seconds (default: 5) */
  maxJitterSeconds?: number;
  /** Whether polling is enabled (default: true) */
  enabled?: boolean;
  /** Whether to start polling immediately (default: true) */
  startImmediately?: boolean;
}

/**
 * Polling state
 */
interface TicketPollingState {
  isPolling: boolean;
  lastPollAt: number | null;
  nextPollAt: number | null;
  pollCount: number;
  startPolling: () => void;
  stopPolling: () => void;
  pollNow: () => Promise<RefreshResult>;
}

/**
 * Add jitter to interval to prevent thundering herd
 */
const addJitter = (baseIntervalMs: number, maxJitterMs: number): number => {
  const jitter = Math.random() * maxJitterMs * 2 - maxJitterMs; // Â±maxJitter
  return Math.max(1000, baseIntervalMs + jitter); // Minimum 1 second
};

/**
 * Hook for automatic ticket polling
 *
 * Integrates with existing refresh infrastructure and handles:
 * - 30s + jitter polling intervals
 * - Background/foreground app state transitions
 * - Screen focus/blur detection
 * - Proper cleanup on unmount
 * - Debounce respect through existing refresh system
 */
export const useTicketPolling = (
  config: TicketPollingConfig = {}
): TicketPollingState => {
  const {
    baseIntervalSeconds = 30,
    maxJitterSeconds = 5,
    enabled = true,
    startImmediately = true,
  } = config;

  const {refreshTickets, canRefreshTickets} = useRefreshControl();

  // Polling state
  const isPollingRef = useRef<boolean>(false);
  const timerRef = useRef<NodeJS.Timeout | null>(null);
  const lastPollAtRef = useRef<number | null>(null);
  const nextPollAtRef = useRef<number | null>(null);
  const pollCountRef = useRef<number>(0);
  const appStateRef = useRef<AppStateStatus>(AppState.currentState);

  /**
   * Calculate next poll time with jitter
   */
  const calculateNextPollTime = useCallback((): number => {
    const baseMs = baseIntervalSeconds * 1000;
    const maxJitterMs = maxJitterSeconds * 1000;
    const intervalWithJitter = addJitter(baseMs, maxJitterMs);
    return Date.now() + intervalWithJitter;
  }, [baseIntervalSeconds, maxJitterSeconds]);

  /**
   * Execute a single poll
   */
  const pollNow = useCallback(async (): Promise<RefreshResult> => {
    if (!enabled) {
      console.log('[TicketPolling] Polling disabled, skipping poll');
      return {
        success: true,
        refreshedApis: [],
        errors: [],
        timestamp: new Date().toISOString(),
      };
    }

    console.log('[TicketPolling] Executing poll now...');

    try {
      lastPollAtRef.current = Date.now();
      pollCountRef.current += 1;

      const result = await refreshTickets();

      console.log('[TicketPolling] Poll completed:', {
        success: result.success,
        refreshedApis: result.refreshedApis.length,
        errors: result.errors.length,
        pollCount: pollCountRef.current,
      });

      return result;
    } catch (error) {
      const errorMessage =
        error instanceof Error ? error.message : 'Ticket polling failed';
      console.error('[TicketPolling] Poll error:', errorMessage);

      return {
        success: false,
        refreshedApis: [],
        errors: [{api: 'tickets' as any, error: errorMessage}],
        timestamp: new Date().toISOString(),
      };
    }
  }, [enabled, refreshTickets]);

  /**
   * Schedule next poll
   */
  const scheduleNextPoll = useCallback(() => {
    if (!enabled || !isPollingRef.current) {
      return;
    }

    // Clear existing timer
    if (timerRef.current) {
      clearTimeout(timerRef.current);
      timerRef.current = null;
    }

    // Calculate delay
    const nextPollTime = calculateNextPollTime();
    const delay = nextPollTime - Date.now();
    nextPollAtRef.current = nextPollTime;

    console.log(
      '[TicketPolling] Scheduling next poll in',
      Math.round(delay / 1000),
      'seconds'
    );

    timerRef.current = setTimeout(async () => {
      // Only poll if we can refresh (respects 60s debounce)
      if (canRefreshTickets() && isPollingRef.current) {
        await pollNow();
        scheduleNextPoll(); // Schedule next poll
      } else {
        console.log(
          '[TicketPolling] Poll skipped (debounce or stopped), rescheduling...'
        );
        scheduleNextPoll(); // Reschedule anyway to keep trying
      }
    }, delay);
  }, [enabled, calculateNextPollTime, canRefreshTickets, pollNow]);

  /**
   * Start polling
   */
  const startPolling = useCallback(() => {
    if (!enabled) {
      console.log('[TicketPolling] Cannot start - polling disabled');
      return;
    }

    if (isPollingRef.current) {
      console.log('[TicketPolling] Already polling, ignoring start request');
      return;
    }

    console.log('[TicketPolling] Starting ticket polling...');
    isPollingRef.current = true;
    scheduleNextPoll();
  }, [enabled, scheduleNextPoll]);

  /**
   * Stop polling
   */
  const stopPolling = useCallback(() => {
    console.log('[TicketPolling] Stopping ticket polling...');
    isPollingRef.current = false;
    nextPollAtRef.current = null;

    if (timerRef.current) {
      clearTimeout(timerRef.current);
      timerRef.current = null;
    }
  }, []);

  /**
   * Handle app state changes
   */
  const handleAppStateChange = useCallback(
    (nextAppState: AppStateStatus) => {
      const previousState = appStateRef.current;
      appStateRef.current = nextAppState;

      console.log(
        '[TicketPolling] App state change:',
        previousState,
        '->',
        nextAppState
      );

      if (
        nextAppState === 'active' &&
        (previousState === 'background' || previousState === 'inactive')
      ) {
        // App came to foreground
        if (enabled && !isPollingRef.current) {
          console.log('[TicketPolling] App foregrounded, resuming polling...');
          // Wait 5 seconds then resume polling
          setTimeout(() => {
            if (appStateRef.current === 'active') {
              startPolling();
            }
          }, 5000);
        }
      } else if (
        (nextAppState === 'background' || nextAppState === 'inactive') &&
        isPollingRef.current
      ) {
        // App went to background
        console.log('[TicketPolling] App backgrounded, pausing polling...');
        stopPolling();
      }
    },
    [enabled, startPolling, stopPolling]
  );

  /**
   * Setup app state listener
   */
  useEffect(() => {
    const subscription = AppState.addEventListener(
      'change',
      handleAppStateChange
    );
    return () => {
      subscription?.remove();
    };
  }, [handleAppStateChange]);

  /**
   * Cleanup on unmount - FORCE stop all timers
   */
  useEffect(() => {
    return () => {
      console.log('[TicketPolling] Cleaning up on unmount - FORCE STOP');
      isPollingRef.current = false;
      nextPollAtRef.current = null;

      if (timerRef.current) {
        clearTimeout(timerRef.current);
        timerRef.current = null;
      }
    };
  }, []);

  return useMemo(
    () => ({
      isPolling: isPollingRef.current,
      lastPollAt: lastPollAtRef.current,
      nextPollAt: nextPollAtRef.current,
      pollCount: pollCountRef.current,
      startPolling,
      stopPolling,
      pollNow,
    }),
    [startPolling, stopPolling, pollNow]
  );
};

export default useTicketPolling;
