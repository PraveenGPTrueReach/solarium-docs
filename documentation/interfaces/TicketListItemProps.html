import React from 'react';
import {View, Text, TouchableOpacity} from 'react-native';
import {StyleSheet} from 'react-native';
import MaterialIcons from 'react-native-vector-icons/MaterialIcons';
import {spacing, borderRadius} from '../../theme/palette';
import {useAppTheme} from '../../theme/ThemeProvider';
import type {TicketListItem} from '../../types/Ticket';
import {getRelativeTime} from '../../utils/dateHelpers';
import {StatusBadge} from '../common/StatusBadge';

export interface TicketListItemProps {
  /** Ticket data to display */
  ticket: TicketListItem;
  /** Callback when item is pressed */
  onPress: (ticketId: string) => void;
  /** Test ID for testing */
  testID?: string;
}

/**
 * TicketListItemComponent
 * Memoized for performance in FlatList
 */
const TicketListItemComponent: React.FC<TicketListItemProps> = React.memo(
  ({ticket, onPress, testID}) => {
    const theme = useAppTheme();
    const relativeTime = getRelativeTime(ticket.createdAt);

    // Accessible label for screen readers
    const accessibleLabel = `Ticket ${ticket.ticketId}, ${ticket.subject}, Status: ${ticket.status}, Created ${relativeTime}`;

    const handlePress = () => {
      onPress(ticket.ticketId);
    };

    return (
      <TouchableOpacity
        style={[styles.container, {backgroundColor: theme.colors.surface}]}
        onPress={handlePress}
        testID={testID}
        accessible={true}
        accessibilityRole="button"
        accessibilityLabel={accessibleLabel}
        accessibilityHint="Tap to view ticket details"
      >
        {/* Ticket Header - ID and Status */}
        <View style={styles.ticketHeader}>
          <Text
            style={[styles.ticketId, {color: theme.colors.primary}]}
            accessible={false}
          >
            {ticket.ticketId}
          </Text>

          <StatusBadge
            status={ticket.status}
            variant="compact"
            testID={`${testID}-status`}
          />
        </View>

        {/* Ticket Subject */}
        <Text
          style={[styles.subject, {color: theme.colors.onSurface}]}
          numberOfLines={2}
          accessible={false}
        >
          {ticket.subject}
        </Text>

        {/* Footer - Lead ID and Created Time */}
        <View style={styles.footer}>
          <View style={styles.leadInfo}>
            <MaterialIcons
              name="assignment"
              size={14}
              color={theme.colors.onSurfaceVariant}
              style={styles.leadIcon}
            />
            <Text
              style={[styles.leadId, {color: theme.colors.onSurfaceVariant}]}
              numberOfLines={1}
              accessible={false}
            >
              {ticket.leadId}
            </Text>
          </View>

          <View style={styles.timeContainer}>
            <MaterialIcons
              name="schedule"
              size={12}
              color={theme.colors.onSurfaceVariant}
              style={styles.timeIcon}
            />
            <Text
              style={[styles.timeText, {color: theme.colors.onSurfaceVariant}]}
              accessible={false}
            >
              {relativeTime}
            </Text>
          </View>
        </View>
      </TouchableOpacity>
    );
  }
);

TicketListItemComponent.displayName = 'TicketListItemComponent';
