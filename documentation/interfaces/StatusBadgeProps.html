import React from 'react';
import {StyleSheet} from 'react-native';
import {Chip} from 'react-native-paper';
import {useAppTheme} from '../../../theme/ThemeProvider';
import type {TicketStatus} from '../../../types/Ticket';

export interface StatusBadgeProps {
  /** Status to display */
  status: TicketStatus;
  /** Badge size variant */
  variant?: 'default' | 'compact';
  /** Test ID for testing */
  testID?: string;
}

/**
 * Get status color configuration based on ticket status
 */
const getStatusConfig = (
  status: TicketStatus,
  theme: any
): {backgroundColor: string; textColor: string} => {
  switch (status) {
    case 'Open':
      return {
        backgroundColor: theme.colors.success || '#4CAF50',
        textColor: '#FFFFFF',
      };
    case 'Closed':
      return {
        backgroundColor: theme.colors.onSurfaceVariant || '#757575',
        textColor: '#FFFFFF',
      };
    default:
      return {
        backgroundColor: theme.colors.primary || '#2E7D32',
        textColor: '#FFFFFF',
      };
  }
};

/**
 * StatusBadge Component
 * Memoized for performance in FlatList
 */
const StatusBadge: React.FC<StatusBadgeProps> = React.memo(
  ({status, variant = 'default', testID}) => {
    const theme = useAppTheme();
    const statusConfig = getStatusConfig(status, theme);

    return (
      <Chip
        style={[
          variant === 'compact' ? styles.compactChip : styles.defaultChip,
          {backgroundColor: statusConfig.backgroundColor},
        ]}
        textStyle={[
          variant === 'compact' ? styles.compactText : styles.defaultText,
          {color: statusConfig.textColor},
        ]}
        compact={variant === 'compact'}
        mode="flat"
        testID={testID}
        accessible={true}
        accessibilityRole="text"
        accessibilityLabel={`Status: ${status}`}
      >
        {status}
      </Chip>
    );
  }
);

StatusBadge.displayName = 'StatusBadge';

/**
 * StatusBadge Styles
 * Styling for status badge component following Material Design patterns
 */
const styles = StyleSheet.create({
  defaultChip: {
    height: 32,
    borderRadius: 16,
  },

  compactChip: {
    height: 28,
    borderRadius: 14,
  },

  defaultText: {
    fontSize: 12,
    fontWeight: '600',
    textTransform: 'uppercase',
    letterSpacing: 0.5,
    lineHeight: 16,
  },

  compactText: {
    fontSize: 10,
    fontWeight: '600',
    textTransform: 'uppercase',
    letterSpacing: 0.3,
    lineHeight: 14,
  },
});

export default StatusBadge;
