export type TicketStatus = 'Open' | 'Closed';

/**
 * Message sender type
 * Based on actual API response structure
 */
export type MessageSender = 'Customer' | 'Admin' | 'KAM';

/**
 * Base complete ticket interface, used in frontend
 * Based on api response structure
 */
export interface Ticket {
  /** Unique ticket identifier */
  ticketId: string;
  /** Associated executed lead identifier */
  leadId: string;
  /** Ticket subject line */
  subject: string;
  /** Initial ticket description */
  description: string;
  /** Current ticket status */
  status: TicketStatus;
  /** All messages in chronological order */
  messages: TicketMessage[];
  /** ISO timestamp when ticket was created */
  createdAt: string;
  /** User ID who created the ticket */
  createdBy: string;
  /** ISO timestamp when ticket was closed (null if open) */
  closedAt: string | null;
}

/**
 * Base ticket interface for list view
 * Based on GET /api/v1/tickets response structure
 */
export interface TicketListItem {
  /** Unique ticket identifier (e.g., "TKT004") */
  ticketId: string;
  /** Associated executed lead identifier */
  leadId: string;
  /** Ticket subject line */
  subject: string;
  /** Current ticket status */
  status: TicketStatus;
  /** ISO timestamp when ticket was created */
  createdAt: string;
  /** User ID who created the ticket */
  createdBy: string;
}

/**
 * Message within a ticket thread
 * Based on messages array structure in GET /api/v1/tickets/{id}
 */
export interface TicketMessage {
  /** Message sender role */
  sender: MessageSender;
  /** ISO timestamp when message was created */
  timestamp: string;
  /** Message content */
  message: string;
}

/**
 * Complete ticket with full details and message thread
 * Based on GET /api/v1/tickets/{id} response structure
 */
export interface GetTicketByIdResponse {
  /** Unique ticket identifier */
  ticketId: string;
  /** Associated executed lead identifier */
  leadId: string;
  /** Ticket subject line */
  subject: string;
  /** Initial ticket description */
  description: string;
  /** Current ticket status */
  status: TicketStatus;
  /** All messages in chronological order */
  messages: TicketMessage[];
  /** ISO timestamp when ticket was created */
  createdAt: string;
  /** ISO timestamp when ticket was closed (null if open) */
  closedAt: string | null;
}

/**
 * Paginated response for ticket list
 * Based on GET /api/v1/tickets response structure
 */
export interface GetTicketsResponse {
  /** Array of tickets */
  items: TicketListItem[];
  /** Total number of tickets available */
  total: number;
  /** Current offset for pagination */
  offset: number;
  /** Current limit for pagination */
  limit: number;
}

/**
 * Query parameters for ticket list endpoint
 * Based on available filters in the API
 */
export interface TicketsQueryParams {
  /** Pagination offset (default: 0) */
  offset?: number;
  /** Items per page (default: 25) */
  limit?: number;
  /** Filter by ticket status */
  status?: TicketStatus;
  /** Filter by channel partner ID */
  cpId?: string;
  /** Filter by customer ID */
  customerId?: string;
  /** Filter by date range */
  dateRange?: string;
}

/**
 * Request payload for creating a new ticket
 * Based on POST /api/v1/tickets request structure
 */
export interface CreateTicketRequest {
  /** Lead ID (must be status=Executed) */
  leadId: string;
  /** Ticket subject */
  subject: string;
  /** Initial ticket description */
  description: string;
}

/**
 * Response for ticket creation
 * Based on POST /api/v1/tickets response structure
 */
export interface CreateTicketResponse {
  /** Newly created ticket ID */
  ticketId: string;
}

/**
 * Request payload for adding a message to ticket
 * Based on POST /api/v1/tickets/{id}/messages request structure
 */
export interface AddMessageRequest {
  /** Message content */
  message: string;
}

/**
 * Redux state interface for tickets
 * Used by the tickets slice for caching
 */
export interface TicketsState {
  /** Ticket list cache */
  list: {
    /** Cached ticket items */
    items: TicketListItem[];
    /** Total count from last fetch */
    total: number;
    /** Last successful fetch timestamp */
    lastFetched: number | null;
    /** Current error message if any */
    error: string | null;
  };
  /** Individual ticket thread cache */
  threads: Record<
    string,
    {
      /** Cached ticket detail data */
      data: GetTicketByIdResponse | null;
      /** Last successful fetch timestamp */
      lastFetched: number | null;
      /** Current error message if any */
      error: string | null;
    }
  >;
}
