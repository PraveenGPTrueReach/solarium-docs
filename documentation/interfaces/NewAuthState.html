import {createSlice, createAsyncThunk, PayloadAction} from '@reduxjs/toolkit';
import {User, LoginResponse} from '../api/endpoints/newAuth';
import {saveJWT, clearJWT, getJWT} from '../storage/secureStorage';

/**
 * Authentication State
 */
export interface NewAuthState {
  isLoggedIn: boolean;
  isLoading: boolean;
  user: User | null;
  token: string | null;
  error: string | null;
}

/**
 * Initial state
 */
const initialState: NewAuthState = {
  isLoggedIn: false,
  isLoading: false,
  user: null,
  token: null,
  error: null,
};

/**
 * Simple auth restoration - just check if token exists
 */
export const restoreAuth = createAsyncThunk(
  'newAuth/restore',
  async (_, {dispatch}) => {
    try {
      const token = await getJWT();
      if (token) {
        // Simple JWT decode to get user info
        const base64Decode = (str: string): string => {
          if (typeof atob !== 'undefined') {
            return atob(str);
          }
          // React Native polyfill
          return Buffer.from(str, 'base64').toString('ascii');
        };

        const payload = JSON.parse(base64Decode(token.split('.')[1]));
        const user: User = {
          id: payload.userId,
          name: payload.name,
          email: payload.email,
          phone: payload.phone,
          role: payload.role,
        };
        dispatch(setAuthState({user, token, isLoggedIn: true}));
        return {user, token};
      }
      return null;
    } catch (error) {
      console.error('[NewAuth] Restore failed:', error);
      await clearJWT(); // Clear invalid token
      return null;
    }
  }
);

/**
 * Login success handler
 */
export const loginSuccess = createAsyncThunk(
  'newAuth/loginSuccess',
  async (loginData: LoginResponse, {dispatch}) => {
    try {
      // Save token to secure storage
      await saveJWT(loginData.accessToken);

      // Update state
      dispatch(
        setAuthState({
          user: loginData.user,
          token: loginData.accessToken,
          isLoggedIn: true,
        })
      );

      return loginData;
    } catch (error) {
      console.error('[NewAuth] Login success handler failed:', error);
      throw error;
    }
  }
);

/**
 * Logout handler
 */
export const logoutUser = createAsyncThunk(
  'newAuth/logout',
  async (_, {dispatch}) => {
    try {
      // Clear secure storage
      await clearJWT();

      // Clear state
      dispatch(clearAuthState());

      return true;
    } catch (error) {
      console.error('[NewAuth] Logout failed:', error);
      // Still clear state even if storage clear fails
      dispatch(clearAuthState());
      return true;
    }
  }
);

/**
 * New Auth Slice
 */
const newAuthSlice = createSlice({
  name: 'newAuth',
  initialState,
  reducers: {
    setLoading: (state, action: PayloadAction<boolean>) => {
      state.isLoading = action.payload;
    },
    setError: (state, action: PayloadAction<string | null>) => {
      state.error = action.payload;
    },
    setAuthState: (
      state,
      action: PayloadAction<{
        user: User;
        token: string;
        isLoggedIn: boolean;
      }>
    ) => {
      state.user = action.payload.user;
      state.token = action.payload.token;
      state.isLoggedIn = action.payload.isLoggedIn;
      state.error = null;
    },
    clearAuthState: state => {
      state.isLoggedIn = false;
      state.user = null;
      state.token = null;
      state.error = null;
    },
  },
  extraReducers: builder => {
    builder
      // Restore auth
      .addCase(restoreAuth.pending, state => {
        state.isLoading = true;
      })
      .addCase(restoreAuth.fulfilled, state => {
        state.isLoading = false;
      })
      .addCase(restoreAuth.rejected, state => {
        state.isLoading = false;
        state.isLoggedIn = false;
        state.user = null;
        state.token = null;
      })

      // Login success
      .addCase(loginSuccess.pending, state => {
        state.isLoading = true;
        state.error = null;
      })
      .addCase(loginSuccess.fulfilled, state => {
        state.isLoading = false;
      })
      .addCase(loginSuccess.rejected, (state, action) => {
        state.isLoading = false;
        state.error = action.error.message || 'Login failed';
      })

      // Logout
      .addCase(logoutUser.fulfilled, state => {
        state.isLoading = false;
      });
  },
});

// Export actions
export const {setLoading, setError, setAuthState, clearAuthState} =
  newAuthSlice.actions;

// Export selectors
export const selectNewAuth = (state: {newAuth: NewAuthState}) => state.newAuth;
export const selectIsLoggedIn = (state: {newAuth: NewAuthState}) =>
  state.newAuth.isLoggedIn;
export const selectUser = (state: {newAuth: NewAuthState}) =>
  state.newAuth.user;
export const selectToken = (state: {newAuth: NewAuthState}) =>
  state.newAuth.token;
export const selectAuthLoading = (state: {newAuth: NewAuthState}) =>
  state.newAuth.isLoading;
export const selectAuthError = (state: {newAuth: NewAuthState}) =>
  state.newAuth.error;

// Export reducer
export default newAuthSlice.reducer;
