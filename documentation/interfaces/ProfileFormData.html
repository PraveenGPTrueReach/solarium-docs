import React, {useCallback, useMemo, useEffect} from 'react';
import {useFormContext} from 'react-hook-form';
import {View, Text, StyleSheet} from 'react-native';
import {Button, TextInput} from 'react-native-paper';
import MaterialIcons from 'react-native-vector-icons/MaterialIcons';
import {useAppTheme} from '../../theme/ThemeProvider';
import type {ProfileData, UpdateProfileRequest} from '../../types/Profile';
import {
  profileValidationRules,
  sanitizeProfileData,
} from '../../utils/profileValidation';
import FormProviderWrapper from './FormProviderWrapper';

/**
 * Profile form data interface (matches form fields)
 */
interface ProfileFormData {
  name: string;
  email: string;
  phone: string; // Read-only, for display only
}

/**
 * Profile form props
 */
export interface ProfileFormProps {
  /** Initial profile data to populate the form */
  initialData?: ProfileData;
  /** Callback when form is submitted with valid data */
  onSubmit: (data: UpdateProfileRequest) => void;
  /** Loading state during form submission */
  isLoading?: boolean;
  /** Error message to display */
  errorMessage?: string;
  /** Test ID for testing purposes */
  testID?: string;
}

/**
 * Profile Form Fields Component (uses useFormContext)
 */
const ProfileFormFields: React.FC<{
  initialData?: ProfileData;
  isLoading: boolean;
  onFormSubmit: () => void; // Add this prop to trigger submission
}> = ({initialData, isLoading, onFormSubmit}) => {
  const theme = useAppTheme();
  const {
    register,
    setValue,
    watch,
    formState: {errors, isDirty},
    reset,
    trigger,
  } = useFormContext<ProfileFormData>();

  // Watch form values for real-time validation
  const nameValue = watch('name');
  const emailValue = watch('email');

  // Register fields with validation rules
  useEffect(() => {
    register('name', profileValidationRules.name);
    register('email', profileValidationRules.email);
    register('phone'); // Read-only field
  }, [register]);

  // Set initial values when data is available
  useEffect(() => {
    if (initialData) {
      setValue('name', initialData.name || '');
      setValue('email', initialData.email || '');
      setValue('phone', initialData.phone || '');
    }
  }, [initialData, setValue]);

  /**
   * Handle name field changes
   */
  const handleNameChange = useCallback(
    (text: string) => {
      console.log('[ProfileForm] Name changed:', text);
      setValue('name', text, {shouldValidate: true, shouldDirty: true});
    },
    [setValue]
  );

  /**
   * Handle email field changes
   */
  const handleEmailChange = useCallback(
    (text: string) => {
      console.log('[ProfileForm] Email changed:', text);
      setValue('email', text, {shouldValidate: true, shouldDirty: true});
    },
    [setValue]
  );

  /**
   * Reset form to initial values
   */
  const handleReset = useCallback(() => {
    console.log('[ProfileForm] Resetting form');
    if (initialData) {
      reset({
        name: initialData.name || '',
        email: initialData.email || '',
        phone: initialData.phone || '',
      });
    }
  }, [reset, initialData]);

  /**
   * Handle form submission
   */
  const handleUpdatePress = useCallback(async () => {
    console.log('[ProfileForm] Update button pressed');
    console.log('[ProfileForm] Current form state:', {
      isDirty,
      errors: Object.keys(errors),
      nameValue,
      emailValue,
    });

    // Trigger validation
    const isValid = await trigger();
    console.log('[ProfileForm] Form validation result:', isValid);

    if (isValid && isDirty) {
      console.log('[ProfileForm] Form is valid and dirty, submitting...');
      onFormSubmit(); // Call the parent's submit function
    } else {
      console.log('[ProfileForm] Form not submitted:', {
        isValid,
        isDirty,
        errors,
      });
    }
  }, [onFormSubmit, isDirty, errors, trigger, nameValue, emailValue]);

  /**
   * Check if form has valid changes
   */
  const hasValidChanges = useMemo(() => {
    const isValid = Object.keys(errors).length === 0;
    console.log('[ProfileForm] Form state check:', {
      isDirty,
      isValid,
      hasValidChanges: isDirty && isValid,
    });
    return isDirty && isValid;
  }, [isDirty, errors]);

  return (
    <View style={styles.fieldsContainer}>
      {/* Phone Number - Read Only Display */}
      <View style={styles.readOnlyFieldContainer}>
        <Text style={[styles.readOnlyLabel, {color: theme.colors.onSurface}]}>
          Phone Number
        </Text>
        <View
          style={[
            styles.readOnlyValueContainer,
            {
              backgroundColor: theme.colors.surfaceVariant,
              borderColor: theme.colors.outline,
            },
          ]}
        >
          <MaterialIcons
            name="phone"
            size={18}
            color={theme.colors.onSurfaceVariant}
            style={styles.readOnlyIcon}
          />
          <Text
            style={[
              styles.readOnlyValue,
              {color: theme.colors.onSurfaceVariant},
            ]}
            accessibilityLabel={`Phone number: ${
              initialData?.phone || 'Not available'
            }`}
            accessibilityRole="text"
          >
            {initialData?.phone || 'Not available'}
          </Text>
        </View>
        <Text
          style={[
            styles.readOnlyHelper,
            {color: theme.colors.onSurfaceVariant},
          ]}
        >
          Phone number cannot be changed
        </Text>
      </View>

      {/* Name Field */}
      <TextInput
        label="Full Name"
        placeholder="Enter your full name"
        value={nameValue || ''}
        onChangeText={handleNameChange}
        error={!!errors.name}
        disabled={isLoading}
        maxLength={100}
        autoCapitalize="words"
        autoCorrect={false}
        style={styles.textInput}
        testID="name-input"
        accessibilityLabel="Full name input field"
        accessibilityHint="Enter your full name, optional field"
      />
      {errors.name && (
        <Text style={[styles.errorText, {color: theme.colors.error}]}>
          {errors.name.message}
        </Text>
      )}
      <Text
        style={[styles.charCountText, {color: theme.colors.onSurfaceVariant}]}
      >
        {(nameValue || '').length}/100 characters
      </Text>

      {/* Email Field */}
      <TextInput
        label="Email Address"
        placeholder="Enter your email address"
        value={emailValue || ''}
        onChangeText={handleEmailChange}
        error={!!errors.email}
        keyboardType="email-address"
        autoCapitalize="none"
        autoCorrect={false}
        disabled={isLoading}
        style={styles.textInput}
        testID="email-input"
        accessibilityLabel="Email address input field"
        accessibilityHint="Enter your email address, optional field"
      />
      {errors.email && (
        <Text style={[styles.errorText, {color: theme.colors.error}]}>
          {errors.email.message}
        </Text>
      )}

      {/* Action Buttons */}
      <View style={styles.actionsContainer}>
        {/* Reset Button */}
        <Button
          mode="outlined"
          onPress={handleReset}
          disabled={isLoading || !isDirty}
          style={[
            styles.resetButton,
            {
              borderColor: theme.colors.outline,
              opacity: !isDirty ? 0.5 : 1,
            },
          ]}
          labelStyle={{color: theme.colors.onSurface}}
          icon={({size, color}) => (
            <MaterialIcons name="refresh" size={size} color={color} />
          )}
          testID="reset-button"
          accessibilityLabel="Reset form to original values"
          accessibilityHint={
            isDirty
              ? 'Reset all changes and restore original values'
              : 'No changes to reset'
          }
        >
          Reset
        </Button>

        {/* Update Button - FIXED */}
        <Button
          mode="contained"
          onPress={handleUpdatePress} // Now calls the proper submit function
          disabled={isLoading || !hasValidChanges}
          loading={isLoading}
          style={[
            styles.updateButton,
            {
              backgroundColor:
                hasValidChanges && !isLoading
                  ? theme.colors.primary
                  : theme.colors.surfaceVariant,
              opacity: !hasValidChanges ? 0.5 : 1,
            },
          ]}
          labelStyle={{
            color:
              hasValidChanges && !isLoading
                ? theme.colors.onPrimary
                : theme.colors.onSurfaceVariant,
          }}
          icon={({size, color}) => (
            <MaterialIcons name="save" size={size} color={color} />
          )}
          testID="update-button"
          accessibilityLabel={
            isLoading
              ? 'Updating profile'
              : hasValidChanges
              ? 'Save profile changes'
              : 'No changes to save'
          }
          accessibilityHint={
            isLoading
              ? 'Please wait, saving your profile changes'
              : !isDirty
              ? 'Make changes to enable this button'
              : !hasValidChanges
              ? 'Please fix form errors first'
              : 'Save your profile changes'
          }
        >
          {isLoading ? 'Updating...' : 'Update Profile'}
        </Button>
      </View>
    </View>
  );
};

/**
 * Profile Form Component
 */
const ProfileForm: React.FC<ProfileFormProps> = ({
  initialData,
  onSubmit,
  isLoading = false,
  errorMessage,
  testID = 'profile-form',
}) => {
  const theme = useAppTheme();

  /**
   * Prepare default values from initial data
   */
  const defaultValues: ProfileFormData = useMemo(() => {
    console.log('[ProfileForm] Setting default values:', initialData);
    return {
      name: initialData?.name || '',
      email: initialData?.email || '',
      phone: initialData?.phone || '', // Read-only field
    };
  }, [initialData]);

  /**
   * Handle form submission with data sanitization
   */
  const handleSubmit = useCallback(
    (formData: ProfileFormData) => {
      console.log('[ProfileForm] Form submitted with data:', formData);

      // Sanitize and prepare data for API
      const sanitizedData = sanitizeProfileData({
        name: formData.name,
        email: formData.email,
      });

      console.log('[ProfileForm] Sanitized data:', sanitizedData);

      // Only submit if there's actually data to update
      if (Object.keys(sanitizedData).length > 0) {
        console.log('[ProfileForm] Calling onSubmit with sanitized data');
        onSubmit(sanitizedData);
      } else {
        console.log('[ProfileForm] No changes to submit');
      }
    },
    [onSubmit]
  );

  /**
   * Handle form submission trigger from child component
   */
  const handleFormSubmit = useCallback(() => {
    console.log('[ProfileForm] handleFormSubmit called');
    // This will be called by the FormProviderWrapper's onSubmit method
  }, []);

  /**
   * Render error message if provided
   */
  const renderErrorMessage = useCallback(() => {
    if (!errorMessage) {
      return null;
    }

    return (
      <View
        style={[
          styles.errorContainer,
          {backgroundColor: theme.colors.errorContainer},
        ]}
      >
        <MaterialIcons
          name="error-outline"
          size={18}
          color={theme.colors.onErrorContainer}
        />
        <Text
          style={[
            styles.errorMessageText,
            {color: theme.colors.onErrorContainer},
          ]}
          accessibilityRole="alert"
        >
          {errorMessage}
        </Text>
      </View>
    );
  }, [errorMessage, theme]);

  return (
    <View style={styles.container} testID={testID}>
      {/* Error Message */}
      {renderErrorMessage()}

      {/* Form */}
      <FormProviderWrapper
        defaultValues={defaultValues}
        formOptions={{
          mode: 'onChange', // Real-time validation
          reValidateMode: 'onChange',
        }}
        onSubmit={handleSubmit} // This will be called when form is submitted
        testID={`${testID}-form-wrapper`}
      >
        {methods => (
          <View style={styles.formContent}>
            {/* Header */}
            <View style={styles.headerContainer}>
              <MaterialIcons
                name="person"
                size={24}
                color={theme.colors.primary}
              />
              <Text
                style={[styles.headerTitle, {color: theme.colors.onSurface}]}
                accessibilityRole="header"
              >
                Profile Information
              </Text>
            </View>

            <Text
              style={[
                styles.headerSubtitle,
                {color: theme.colors.onSurfaceVariant},
              ]}
            >
              Update your personal information. All fields are optional.
            </Text>

            {/* Form Fields */}
            <ProfileFormFields
              initialData={initialData}
              isLoading={isLoading}
              onFormSubmit={methods.handleSubmit(handleSubmit)} // Pass the actual submit function
            />
          </View>
        )}
      </FormProviderWrapper>
    </View>
  );
};
