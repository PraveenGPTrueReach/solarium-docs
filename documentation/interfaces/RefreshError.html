export interface RefreshState {
  isRefreshing: boolean;
  lastRefreshAt: string | null;
  refreshError: string | null;
  refreshCount: number;
  refreshProgress: RefreshProgress;
}

export interface RefreshProgress {
  current: number;
  total: number;
  currentStep: string;
}

export interface RefreshOptions {
  /**
   * Whether to show loading indicators during refresh
   * @default true
   */
  showLoading?: boolean;

  /**
   * Whether to include user-specific data in refresh
   * @default true
   */
  includeUserData?: boolean;

  /**
   * Specific APIs to refresh (if not provided, refreshes all)
   */
  apis?: RefreshableApi[];

  /**
   * Whether this is a background refresh
   * @default false
   */
  isBackground?: boolean;
}

export enum RefreshableApi {
  AUTH = 'auth',
  SERVICES = 'services',
  LEADS = 'leads',
  SINGLE_LEAD = 'single_lead',
  QUOTATIONS = 'quotations',
  SINGLE_QUOTATION = 'single_quotation',
  KYC = 'kyc',
  NOTIFICATIONS = 'notifications',
  TICKETS = 'tickets',
  // Add more APIs here as they're implemented
}

export interface RefreshResult {
  success: boolean;
  refreshedApis: RefreshableApi[];
  errors: RefreshError[];
  timestamp: string;
}

export interface RefreshError {
  api: RefreshableApi;
  error: string;
  code?: string;
}

export interface RefreshHookReturn {
  // Actions
  refreshAll: (options?: RefreshOptions) => Promise<RefreshResult>;
  refreshSpecific: (
    apis: RefreshableApi[],
    options?: RefreshOptions
  ) => Promise<RefreshResult>;

  // State
  isRefreshing: boolean;
  lastRefreshAt: string | null;
  refreshError: string | null;
  refreshCount: number;
  refreshProgress: RefreshProgress;

  // Utilities
  getFormattedLastRefresh: () => string;
  canRefresh: boolean;
  clearError: () => void;

  // Leads-specific with debounce
  refreshLeads: () => Promise<RefreshResult>;
  canRefreshLeads: () => boolean;

  // Single lead-specific with debounce
  refreshSingleLead: (leadId: string) => Promise<RefreshResult>;
  canRefreshSingleLead: () => boolean;

  // Quotations-specific with debounce
  refreshQuotations: () => Promise<RefreshResult>;
  canRefreshQuotations: () => boolean;

  // Single quotation-specific with debounce
  refreshSingleQuotation: (quotationId: string) => Promise<RefreshResult>;
  canRefreshSingleQuotation: () => boolean;

  // KYC-specific with debounce
  refreshKyc: () => Promise<RefreshResult>;
  canRefreshKyc: () => boolean;

  // Notifications-specific with debounce
  refreshNotifications: () => Promise<RefreshResult>;
  canRefreshNotifications: () => boolean;

  // Tickets-specific with debounce
  refreshTickets: () => Promise<RefreshResult>;
  canRefreshTickets: () => boolean;
}
