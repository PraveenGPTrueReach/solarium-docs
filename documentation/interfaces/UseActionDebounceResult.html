import {useState, useCallback, useEffect, useRef} from 'react';
import {
  isOutsideDebounceWindow,
  DEBOUNCE_WINDOWS,
} from '../utils/debounceWindow';

/**
 * Return type for useActionDebounce hook
 */
export interface UseActionDebounceResult {
  /** Whether the action is currently blocked due to debounce */
  isBlocked: boolean;
  /** Function to trigger the action and start debounce timer */
  triggerAction: () => void;
  /** Remaining seconds in debounce window (0 if not blocked) */
  remainingSeconds: number;
}

/**
 * Options for configuring the action debounce hook
 */
export interface UseActionDebounceOptions {
  /** Custom debounce window in seconds (default: DEBOUNCE_WINDOWS.QUOTE_ACTION) */
  debounceSeconds?: number;
}

/**
 * Global map to store last action timestamps per quotation ID
 * This ensures debounce state persists across component re-renders and is independent per quotation
 */
const quotationActionTimestamps = new Map<string, number>();

/**
 * Action Debounce Hook
 *
 * Provides debounced action handling with automatic state management for quotation actions.
 * Each quotation has independent debounce state, preventing rapid successive actions while
 * allowing actions on different quotations simultaneously.
 *
 * The hook uses existing debounceWindow utilities for time calculations and maintains
 * a global Map for timestamp storage to ensure consistency across component lifecycle.
 *
 * @param quotationId - Unique identifier for the quotation (used as debounce key)
 * @param options - Configuration options for debounce behavior
 * @returns Object containing isBlocked state, triggerAction function, and remaining seconds
 *
 * @example
 * ```typescript
 * const { isBlocked, triggerAction, remainingSeconds } = useActionDebounce('quote-123');
 *
 * const handleAccept = () => {
 *   if (isBlocked) return;
 *   triggerAction(); // Starts 5-second debounce
 *   // Perform actual accept logic
 * };
 * ```
 */
export const useActionDebounce = (
  quotationId: string,
  options: UseActionDebounceOptions = {}
): UseActionDebounceResult => {
  const {debounceSeconds = DEBOUNCE_WINDOWS.QUOTE_ACTION} = options;

  // State to track if action is currently blocked
  const [isBlocked, setIsBlocked] = useState<boolean>(false);
  const [remainingSeconds, setRemainingSeconds] = useState<number>(0);

  // Timer ref for cleanup
  const timerRef = useRef<NodeJS.Timeout | null>(null);
  const intervalRef = useRef<NodeJS.Timeout | null>(null);

  /**
   * Update the blocked state based on current timestamp
   */
  const updateBlockedState = useCallback(() => {
    const lastActionTime = quotationActionTimestamps.get(quotationId) || null;
    const isActionAllowed = isOutsideDebounceWindow(
      lastActionTime,
      debounceSeconds
    );

    const blocked = !isActionAllowed;
    setIsBlocked(blocked);

    if (blocked && lastActionTime) {
      const now = Date.now();
      const debounceMs = debounceSeconds * 1000;
      const timeSinceLastAction = now - lastActionTime;
      const remainingMs = debounceMs - timeSinceLastAction;
      const remainingSecs = Math.max(0, Math.ceil(remainingMs / 1000));
      setRemainingSeconds(remainingSecs);
    } else {
      setRemainingSeconds(0);
    }
  }, [quotationId, debounceSeconds]);

  /**
   * Trigger the action and start debounce timer
   */
  const triggerAction = useCallback(() => {
    const now = Date.now();

    // Check if action is currently allowed
    const lastActionTime = quotationActionTimestamps.get(quotationId) || null;
    if (!isOutsideDebounceWindow(lastActionTime, debounceSeconds)) {
      // Action is still blocked, don't trigger
      return;
    }

    // Record the action timestamp
    quotationActionTimestamps.set(quotationId, now);

    // Update state immediately
    setIsBlocked(true);
    setRemainingSeconds(debounceSeconds);

    // Clear existing timers
    if (timerRef.current) {
      clearTimeout(timerRef.current);
    }
    if (intervalRef.current) {
      clearInterval(intervalRef.current);
    }

    // Start countdown interval
    intervalRef.current = setInterval(() => {
      updateBlockedState();
    }, 1000);

    // Set timer to unblock after debounce period
    timerRef.current = setTimeout(() => {
      setIsBlocked(false);
      setRemainingSeconds(0);

      if (intervalRef.current) {
        clearInterval(intervalRef.current);
        intervalRef.current = null;
      }
    }, debounceSeconds * 1000);
  }, [quotationId, debounceSeconds, updateBlockedState]);

  // Initialize state on mount and when quotationId changes
  useEffect(() => {
    updateBlockedState();
  }, [updateBlockedState]);

  // Cleanup timers on unmount
  useEffect(() => {
    return () => {
      if (timerRef.current) {
        clearTimeout(timerRef.current);
      }
      if (intervalRef.current) {
        clearInterval(intervalRef.current);
      }
    };
  }, []);

  return {
    isBlocked,
    triggerAction,
    remainingSeconds,
  };
};

export default useActionDebounce;
