import {createApi} from '@reduxjs/toolkit/query/react';
import {ServiceFetchError} from '../../errors';
import type {
  ValidationError,
  Lead,
  CreateLeadRequest,
  LeadResponse,
  LeadStatus,
} from '../../types';
import {
  isValidationError,
  isServerError,
  isApiErrorResponse,
} from '../../types';
import type {PaginatedRequest, PaginatedResponse} from '../../utils';
import {calculatePayloadSize, isPayloadTooLarge} from '../../utils';
import {
  baseQuery,
  transformResponse,
  transformError,
  provideTags,
} from '../baseQuery';

/**
 * Request Types
 */
export interface GetLeadsRequest extends PaginatedRequest {
  status?: LeadStatus;
  priority?: 'Low' | 'Medium' | 'High';
  source?: 'App' | 'Website' | 'Referral' | 'Direct';
  customerId?: string;
  fromDate?: string;
  toDate?: string;
}

/**
 * Enhanced error transformation for leads API
 */
const transformLeadError = (error: any): ServiceFetchError => {
  // Check for validation errors
  if (isValidationError(error)) {
    const fieldErrors = Object.entries(error.errors)
      .map(([field, messages]) => `${field}: ${messages.join(', ')}`)
      .join('; ');

    return new ServiceFetchError(
      `Validation failed: ${fieldErrors}`,
      'LEAD_VALIDATION',
      'VALIDATION_ERROR',
      {httpStatus: error.status}
    );
  }

  // Check for server errors
  if (isServerError(error)) {
    return new ServiceFetchError(
      error.message,
      'LEAD_SERVER_ERROR',
      'SERVER_ERROR',
      {httpStatus: error.status}
    );
  }

  // Check for API error responses
  if (isApiErrorResponse(error)) {
    return new ServiceFetchError(error.error, 'LEAD_API_ERROR', 'API_ERROR', {
      httpStatus: 500,
    });
  }

  // Fallback to standard error transformation
  const baseError = transformError(error);
  return new ServiceFetchError(
    baseError.message,
    'LEAD_UNKNOWN_ERROR',
    'UNKNOWN_ERROR',
    {httpStatus: baseError.code}
  );
};

/**
 * Leads API Definition
 */
export const leadsApi = createApi({
  reducerPath: 'leadsApi',
  baseQuery,
  tagTypes: ['Lead', 'LeadsList'],
  endpoints: builder => ({
    /**
     * Get leads with filters
     */
    getLeads: builder.query<PaginatedResponse<Lead>, GetLeadsRequest>({
      query: (params = {}) => {
        const {page = 1, limit = 20, ...filters} = params;
        return {
          url: 'leads',
          params: {page, limit, ...filters},
        };
      },
      // ✅ Custom transformResponse that maps API structure to PaginatedResponse
      transformResponse: (response: any): PaginatedResponse<Lead> => {
        // Handle the standard {success, data} structure first
        let apiData;
        if (response.success && response.data) {
          apiData = response.data;
        } else {
          throw new Error('Invalid leads response format');
        }

        // Transform to match PaginatedResponse interface
        return {
          data: apiData.items || [], // Extract items array
          pagination: {
            page: Math.floor(apiData.offset / apiData.limit) + 1,
            limit: apiData.limit,
            total: apiData.total,
            totalPages: Math.ceil(apiData.total / apiData.limit),
            hasNext: apiData.offset + apiData.limit < apiData.total,
            hasPrev: apiData.offset > 0,
          },
        };
      },
      transformErrorResponse: transformLeadError,
      providesTags: result => [
        ...provideTags('Lead', result?.data),
        {type: 'LeadsList' as const, id: 'LIST'},
      ],
    }),

    /**
     * Get leads for the current customer
     * Returns up to 100 leads for the authenticated customer
     */
    getCustomerLeads: builder.query<Lead[], void>({
      query: () => {
        const params = {
          limit: 100,
          sortBy: 'updatedAt',
          sortOrder: 'desc',
        };

        // Validate payload size (should be small for query params)
        if (isPayloadTooLarge(params, 1)) {
          // 1KB limit for query params
          throw new Error('Query parameters too large');
        }

        return {
          url: 'leads',
          params,
        };
      },
      transformResponse: (response: any) => {
        // Handle the actual API response structure
        let leads: Lead[] = [];

        if (response.success && response.data && response.data.items) {
          // API returns: { success: true, data: { items: [...], total, offset, limit } }
          leads = response.data.items;
        } else if (response.data && Array.isArray(response.data)) {
          // Fallback: direct array response
          leads = response.data;
        } else if (Array.isArray(response)) {
          // Fallback: direct array
          leads = response;
        }

        // Transform API response to match our Lead interface
        const transformedLeads = leads.map((lead: any) => ({
          ...lead,
          id: lead.leadId || lead.id, // Map leadId to id
          customerId: lead.customerId || `CUST-${lead.phone?.slice(-4)}`, // Generate if missing
          customer: lead.customer || {
            id: lead.customerId || `CUST-${lead.phone?.slice(-4)}`,
            name: lead.customerName,
            phone: lead.phone,
          },
        }));

        // Client-side enforcement of 100 item limit
        return transformedLeads.slice(0, 100);
      },
      transformErrorResponse: transformLeadError,
      providesTags: result => [
        ...provideTags('Lead', result),
        {type: 'LeadsList' as const, id: 'CUSTOMER'},
      ],
    }),

    /**
     * Get lead by ID
     */
    getLeadById: builder.query<Lead, string>({
      query: leadId => `leads/${leadId}`,
      transformResponse: transformResponse<Lead>,
      transformErrorResponse: transformLeadError,
      providesTags: (result, error, leadId) => [
        {type: 'Lead' as const, id: leadId},
      ],
    }),

    /**
     * Create new lead - Real API implementation
     */
    createLead: builder.mutation<Lead, CreateLeadRequest>({
      query: data => ({
        url: 'leads',
        method: 'POST',
        body: data,
      }),
      transformResponse: (response: LeadResponse) => {
        // Handle backend response format
        if (response.success && response.data) {
          return response.data;
        }
        // Direct Lead object response
        return response.data as Lead;
      },
      transformErrorResponse: transformLeadError,
      invalidatesTags: [
        {type: 'LeadsList' as const, id: 'LIST'},
        {type: 'LeadsList' as const, id: 'CUSTOMER'},
      ],
      onQueryStarted: async (payload, {queryFulfilled}) => {
        try {
          // Client-side payload size validation
          if (isPayloadTooLarge(payload, 100)) {
            const actualSize = calculatePayloadSize(payload);
            const validationError: ValidationError = {
              status: 400,
              message: 'Request payload too large',
              errors: {
                payload: [
                  `Payload size ${Math.round(
                    actualSize / 1024
                  )}KB exceeds 100KB limit`,
                ],
              },
            };
            console.warn('[CreateLead] Payload too large:', {
              size: `${Math.round(actualSize / 1024)}KB`,
              limit: '100KB',
              payload: `${JSON.stringify(payload).substring(0, 200)}...`,
            });

            throw new ServiceFetchError(
              validationError.message,
              'CREATE_LEAD',
              'PAYLOAD_TOO_LARGE',
              {httpStatus: 400}
            );
          }

          // Wait for the query to complete
          await queryFulfilled;
        } catch (error) {
          // Error handling is done by transformErrorResponse
          console.error('[CreateLead] Mutation failed:', error);
        }
      },
    }),
  }),
});

  useCreateLeadMutation,
