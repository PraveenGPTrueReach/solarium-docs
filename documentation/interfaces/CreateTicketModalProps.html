import React, {useState, useCallback, useMemo, useEffect} from 'react';
import {View, StyleSheet, ScrollView} from 'react-native';
import {
  Modal,
  Portal,
  Card,
  Text,
  Button,
  TextInput,
  Divider,
} from 'react-native-paper';
import MaterialIcons from 'react-native-vector-icons/MaterialIcons';
import {useGetLeadsQuery} from '../../api/endpoints/leads';
import {useCreateTicketMutation} from '../../api/endpoints/tickets';
import {useAppTheme} from '../../theme/ThemeProvider';
import {ErrorMessage} from '../common';
import {Dropdown, type DropdownOption} from '../common';

/**
 * Create Ticket Modal Props
 */
export interface CreateTicketModalProps {
  visible: boolean;
  onClose: () => void;
  onSuccess?: (ticketId: string) => void;
  prefilledLeadId?: string; // For LeadDetail screen usage
  testID?: string;
}

/**
 * Create Ticket Modal Component
 */
const CreateTicketModal: React.FC<CreateTicketModalProps> = ({
  visible,
  onClose,
  onSuccess,
  prefilledLeadId,
  testID = 'create-ticket-modal',
}) => {
  const theme = useAppTheme();

  // Form state
  const [selectedLeadId, setSelectedLeadId] = useState<string>('');
  const [subject, setSubject] = useState<string>('');
  const [description, setDescription] = useState<string>('');
  const [error, setError] = useState<string | null>(null);

  // Form validation errors
  const [fieldErrors, setFieldErrors] = useState<{
    leadId?: string;
    subject?: string;
    description?: string;
  }>({});

  // API hooks
  const {
    data: leadsResponse,
    isLoading: isLoadingLeads,
    error: leadsError,
  } = useGetLeadsQuery({
    status: 'Executed',
    limit: 100,
  });

  const [createTicket, {isLoading: isCreatingTicket}] =
    useCreateTicketMutation();

  // Reset form when modal visibility changes
  useEffect(() => {
    if (visible) {
      setError(null);
      setFieldErrors({});
      if (prefilledLeadId) {
        setSelectedLeadId(prefilledLeadId);
      } else {
        setSelectedLeadId('');
      }
      setSubject('');
      setDescription('');
    }
  }, [visible, prefilledLeadId]);

  /**
   * Convert leads to dropdown options
   */
  const leadOptions: DropdownOption[] = useMemo(() => {
    // Filter for executed leads on client-side
    const executedLeads = leadsResponse?.data || [];

    return executedLeads.map((lead: any) => ({
      value: lead.leadId,
      label: lead.leadId,
      sublabel: `Services:\n ${lead.services.join(',\n ')}`,
    }));
  }, [leadsResponse]);

  /**
   * Validate form fields
   */
  const validateForm = useCallback((): boolean => {
    const errors: typeof fieldErrors = {};

    // Validate leadId
    if (!selectedLeadId.trim()) {
      errors.leadId = 'Please select a lead';
    }

    // Validate subject
    if (!subject.trim()) {
      errors.subject = 'Subject is required';
    } else if (subject.length > 200) {
      errors.subject = 'Subject must be less than 200 characters';
    }

    // Validate description
    if (!description.trim()) {
      errors.description = 'Description is required';
    } else if (description.length > 2000) {
      errors.description = 'Description must be less than 2000 characters';
    }

    setFieldErrors(errors);
    return Object.keys(errors).length === 0;
  }, [selectedLeadId, subject, description]);

  /**
   * Handle form submission
   */
  const handleSubmit = useCallback(async () => {
    setError(null);

    if (!validateForm()) {
      return;
    }

    try {
      const leadIdToUse = prefilledLeadId || selectedLeadId;
      console.log('[CreateTicketModal] Creating ticket:', {
        leadId: leadIdToUse,
        subject: subject.trim(),
        descriptionLength: description.trim().length,
      });

      const result = await createTicket({
        leadId: leadIdToUse,
        subject: subject.trim(),
        description: description.trim(),
      }).unwrap();

      console.log('[CreateTicketModal] Ticket created successfully:', result);

      // Call success callback with ticket ID
      if (onSuccess && result.ticketId) {
        onSuccess(result.ticketId);
      }

      // Close modal
      onClose();
    } catch (createError: any) {
      console.error('[CreateTicketModal] Create ticket error:', createError);

      let errorMessage = 'Failed to create ticket. Please try again.';

      if (createError?.data?.error) {
        errorMessage = createError.data.error;
      } else if (createError?.message) {
        errorMessage = createError.message;
      }

      setError(errorMessage);
    }
  }, [
    selectedLeadId,
    subject,
    description,
    createTicket,
    onSuccess,
    onClose,
    validateForm,
    prefilledLeadId,
  ]);

  /**
   * Handle lead selection
   */
  const handleLeadSelect = useCallback(
    (leadId: string) => {
      setSelectedLeadId(leadId);
      if (fieldErrors.leadId) {
        setFieldErrors(prev => ({...prev, leadId: undefined}));
      }
    },
    [fieldErrors.leadId]
  );

  /**
   * Handle subject change
   */
  const handleSubjectChange = useCallback(
    (text: string) => {
      setSubject(text);
      if (fieldErrors.subject) {
        setFieldErrors(prev => ({...prev, subject: undefined}));
      }
    },
    [fieldErrors.subject]
  );

  /**
   * Handle description change
   */
  const handleDescriptionChange = useCallback(
    (text: string) => {
      setDescription(text);
      if (fieldErrors.description) {
        setFieldErrors(prev => ({...prev, description: undefined}));
      }
    },
    [fieldErrors.description]
  );

  /**
   * Clear error message
   */
  const clearError = useCallback(() => {
    setError(null);
  }, []);

  return (
    <Portal>
      <Modal
        visible={visible}
        onDismiss={onClose}
        contentContainerStyle={styles.modalContainer}
        testID={testID}
      >
        <Card style={[styles.card, {backgroundColor: theme.colors.surface}]}>
          <ScrollView showsVerticalScrollIndicator={false}>
            <Card.Content>
              <Text
                variant="headlineSmall"
                style={[styles.title, {color: theme.colors.onSurface}]}
                accessibilityRole="header"
                accessibilityLabel="Create new support ticket"
              >
                Create Support Ticket
              </Text>

              <Text
                variant="bodyMedium"
                style={[
                  styles.subtitle,
                  {color: theme.colors.onSurfaceVariant},
                ]}
              >
                Create a new support ticket for your executed lead
              </Text>

              {error && (
                <ErrorMessage
                  message={error}
                  type="error"
                  visible={true}
                  dismissible={true}
                  onDismiss={clearError}
                  style={styles.errorMessage}
                  testID="create-ticket-error"
                />
              )}

              <View style={styles.formContainer}>
                {/* Lead Selection */}
                {prefilledLeadId ? (
                  <View style={{marginBottom: 16}}>
                    <Text style={{fontWeight: 'bold', marginBottom: 4}}>
                      Lead ID
                    </Text>
                    <Text style={{color: theme.colors.onSurfaceVariant}}>
                      {prefilledLeadId}
                    </Text>
                  </View>
                ) : (
                  <Dropdown
                    label="Select Lead *"
                    placeholder="Choose an executed lead..."
                    value={selectedLeadId}
                    options={leadOptions}
                    onSelect={handleLeadSelect}
                    error={fieldErrors.leadId}
                    loading={isLoadingLeads}
                    disabled={!!prefilledLeadId}
                    testID="lead-dropdown"
                  />
                )}

                {(leadsError ||
                  (leadOptions.length === 0 && !isLoadingLeads)) && (
                  <Text
                    variant="bodySmall"
                    style={[styles.helperText, {color: theme.colors.error}]}
                  >
                    {leadsError
                      ? 'Failed to load leads. Please try again.'
                      : 'No executed leads found. Please ensure you have leads with "Executed" status.'}
                  </Text>
                )}

                {/* Subject Field */}
                <TextInput
                  label="Subject *"
                  placeholder="Enter ticket subject..."
                  value={subject}
                  onChangeText={handleSubjectChange}
                  error={!!fieldErrors.subject}
                  maxLength={200}
                  disabled={isCreatingTicket}
                  style={styles.textInput}
                  testID="subject-input"
                />
                {fieldErrors.subject && (
                  <Text
                    variant="bodySmall"
                    style={[styles.fieldErrorText, {color: theme.colors.error}]}
                  >
                    {fieldErrors.subject}
                  </Text>
                )}
                <Text
                  variant="bodySmall"
                  style={[
                    styles.charCountText,
                    {color: theme.colors.onSurfaceVariant},
                  ]}
                >
                  {subject.length}/200 characters
                </Text>

                {/* Description Field */}
                <TextInput
                  label="Description *"
                  placeholder="Describe your issue in detail..."
                  value={description}
                  onChangeText={handleDescriptionChange}
                  error={!!fieldErrors.description}
                  multiline
                  numberOfLines={4}
                  maxLength={2000}
                  disabled={isCreatingTicket}
                  style={[styles.textInput, styles.descriptionInput]}
                  testID="description-input"
                />
                {fieldErrors.description && (
                  <Text
                    variant="bodySmall"
                    style={[styles.fieldErrorText, {color: theme.colors.error}]}
                  >
                    {fieldErrors.description}
                  </Text>
                )}
                <Text
                  variant="bodySmall"
                  style={[
                    styles.charCountText,
                    {color: theme.colors.onSurfaceVariant},
                  ]}
                >
                  {description.length}/2000 characters
                </Text>
              </View>

              <Divider style={styles.divider} />

              <Text
                variant="bodySmall"
                style={[
                  styles.helpText,
                  {color: theme.colors.onSurfaceVariant},
                ]}
              >
                * All fields are required{'\n'}
                Once created, you can track and communicate through the ticket
                thread.
              </Text>
            </Card.Content>

            <Card.Actions style={styles.actions}>
              <Button
                mode="text"
                onPress={onClose}
                disabled={isCreatingTicket}
                accessibilityLabel="Cancel ticket creation"
                accessibilityRole="button"
                testID="cancel-button"
              >
                Cancel
              </Button>

              <Button
                mode="contained"
                onPress={handleSubmit}
                disabled={isCreatingTicket}
                loading={isCreatingTicket}
                icon={({size, color}) => (
                  <MaterialIcons name="add" size={size} color={color} />
                )}
                accessibilityLabel={
                  isCreatingTicket ? 'Creating ticket' : 'Create ticket'
                }
                accessibilityRole="button"
                testID="create-button"
              >
                {isCreatingTicket ? 'Creating...' : 'Create Ticket'}
              </Button>
            </Card.Actions>
          </ScrollView>
        </Card>
      </Modal>
    </Portal>
  );
};

const styles = StyleSheet.create({
  modalContainer: {
    margin: 20,
    maxHeight: '90%',
  },
  card: {
    borderRadius: 12,
    elevation: 8,
  },
  title: {
    textAlign: 'center',
    marginBottom: 8,
    fontWeight: 'bold',
  },
  subtitle: {
    textAlign: 'center',
    marginBottom: 24,
  },
  errorMessage: {
    marginBottom: 16,
  },
  formContainer: {
    marginBottom: 16,
  },
  textInput: {
    marginBottom: 4,
    backgroundColor: 'transparent',
  },
  descriptionInput: {
    minHeight: 100,
  },
  fieldErrorText: {
    marginLeft: 16,
    marginTop: 4,
    marginBottom: 4,
  },
  charCountText: {
    textAlign: 'right',
    marginBottom: 16,
    marginRight: 16,
  },
  helperText: {
    marginLeft: 16,
    marginBottom: 16,
  },
  divider: {
    marginVertical: 16,
  },
  helpText: {
    textAlign: 'center',
    lineHeight: 18,
  },
  actions: {
    justifyContent: 'space-between',
    paddingTop: 8,
    paddingHorizontal: 16,
  },
});

export default CreateTicketModal;
