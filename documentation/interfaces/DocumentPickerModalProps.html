import * as DocumentPicker from '@react-native-documents/picker';
import React, {useState, useCallback} from 'react';
import {
  View,
  StyleSheet,
  Alert,
  Platform,
  PermissionsAndroid,
} from 'react-native';
import {
  ImagePickerResponse,
  launchImageLibrary,
  launchCamera,
  MediaType,
  ImageLibraryOptions,
  CameraOptions,
} from 'react-native-image-picker';
import {Modal, Portal, Card, Text, Button, Divider} from 'react-native-paper';
import MaterialIcons from 'react-native-vector-icons/MaterialIcons';
import {ValidationError} from '../../errors/ValidationError';
import {useAppTheme} from '../../theme/ThemeProvider';
import type {RNFile} from '../../types/File';
import {validateMimeType} from '../../utils/kycValidation';
import {ErrorMessage} from '../common';

/**
 * Document Picker Modal Props
 */
export interface DocumentPickerModalProps {
  visible: boolean;
  onClose: () => void;
  onFilePicked: (file: RNFile) => void;
  testID?: string;
}

/**
 * Allowed MIME types for validation
 */
const ALLOWED_MIME_TYPES = ['image/jpeg', 'image/png', 'application/pdf'];

/**
 * Maximum file size (10MB in bytes)
 */
const MAX_FILE_SIZE = 10 * 1024 * 1024;

/**
 * Document Picker Modal Component
 * Provides options to select files from gallery, camera, or file browser
 */
const DocumentPickerModal: React.FC<DocumentPickerModalProps> = ({
  visible,
  onClose,
  onFilePicked,
  testID = 'document-picker-modal',
}) => {
  const theme = useAppTheme();
  const [error, setError] = useState<string | null>(null);
  const [isProcessing, setIsProcessing] = useState(false);

  /**
   * Clear error when modal visibility changes
   */
  React.useEffect(() => {
    if (visible) {
      setError(null);
    }
  }, [visible]);

  /**
   * Validate selected file
   */
  const validateFile = useCallback((file: RNFile): boolean => {
    try {
      // Validate MIME type
      validateMimeType(file.type);

      // Validate file size
      if (file.size > MAX_FILE_SIZE) {
        throw new ValidationError(
          'File size must be less than 10MB',
          'FILE_SIZE_EXCEEDED',
          'fileSize',
          file.size
        );
      }

      return true;
    } catch (validationError) {
      if (validationError instanceof ValidationError) {
        setError(validationError.message);
      } else {
        setError('File validation failed. Please try another file.');
      }
      return false;
    }
  }, []);

  /**
   * Process selected file and validate
   */
  const processSelectedFile = useCallback(
    (file: RNFile) => {
      setIsProcessing(true);

      try {
        if (validateFile(file)) {
          console.log('[DocumentPickerModal] File validated successfully:', {
            name: file.name,
            type: file.type,
            size: file.size,
          });

          onFilePicked(file);
          onClose();
        }
      } catch (error) {
        console.error('[DocumentPickerModal] File processing error:', error);
        setError('Failed to process selected file. Please try again.');
      } finally {
        setIsProcessing(false);
      }
    },
    [validateFile, onFilePicked, onClose]
  );

  /**
   * Handle gallery selection
   */
  const handleGalleryPick = useCallback(() => {
    setError(null);

    const options: ImageLibraryOptions = {
      mediaType: 'photo' as MediaType,
      quality: 1,
      includeBase64: false,
    };

    console.log('[DocumentPickerModal] Opening gallery...');

    launchImageLibrary(options, (response: ImagePickerResponse) => {
      console.log('[DocumentPickerModal] Gallery response:', response);

      if (response.didCancel) {
        console.log('[DocumentPickerModal] Gallery selection cancelled');
        return;
      }

      if (response.errorMessage) {
        console.error(
          '[DocumentPickerModal] Gallery error:',
          response.errorMessage
        );
        setError('Failed to select image from gallery. Please try again.');
        return;
      }

      const asset = response.assets?.[0];
      if (!asset || !asset.uri) {
        console.error('[DocumentPickerModal] No asset or URI found');
        setError('Failed to get image from gallery. Please try again.');
        return;
      }

      console.log('[DocumentPickerModal] Gallery asset:', asset);

      const file: RNFile = {
        uri: asset.uri,
        name: asset.fileName || `image_${Date.now()}.jpg`,
        type: asset.type || 'image/jpeg',
        size: asset.fileSize || 0,
      };

      processSelectedFile(file);
    });
  }, [processSelectedFile]);

  /**
   * Handle camera capture
   */
  const handleCameraCapture = useCallback(async () => {
    setError(null);

    console.log('[DocumentPickerModal] Opening camera...');

    // Request camera permission on Android
    if (Platform.OS === 'android') {
      try {
        // Check if permission is already granted
        const hasPermission = await PermissionsAndroid.check(
          PermissionsAndroid.PERMISSIONS.CAMERA
        );

        if (!hasPermission) {
          console.log('[DocumentPickerModal] Requesting camera permission...');
          const granted = await PermissionsAndroid.request(
            PermissionsAndroid.PERMISSIONS.CAMERA,
            {
              title: 'Camera Permission',
              message:
                'This app needs access to camera to take photos for KYC documents.',
              buttonNeutral: 'Ask Me Later',
              buttonNegative: 'Cancel',
              buttonPositive: 'OK',
            }
          );

          console.log('[DocumentPickerModal] Permission result:', granted);

          if (granted !== PermissionsAndroid.RESULTS.GRANTED) {
            setError('Camera permission is required to take photos.');
            return;
          }
        }
      } catch (error) {
        console.error('[DocumentPickerModal] Camera permission error:', error);
        setError('Failed to request camera permission.');
        return;
      }
    }

    const options: CameraOptions = {
      mediaType: 'photo' as MediaType,
      quality: 1,
      includeBase64: false,
    };

    launchCamera(options, (response: ImagePickerResponse) => {
      console.log('[DocumentPickerModal] Camera response:', response);

      if (response.didCancel) {
        console.log('[DocumentPickerModal] Camera capture cancelled');
        return;
      }

      if (response.errorMessage) {
        console.error(
          '[DocumentPickerModal] Camera error:',
          response.errorMessage
        );
        setError('Failed to capture photo. Please try again.');
        return;
      }

      const asset = response.assets?.[0];
      if (!asset || !asset.uri) {
        console.error('[DocumentPickerModal] No asset or URI found');
        setError('Failed to capture photo. Please try again.');
        return;
      }

      console.log('[DocumentPickerModal] Camera asset:', asset);

      const file: RNFile = {
        uri: asset.uri,
        name: asset.fileName || `photo_${Date.now()}.jpg`,
        type: asset.type || 'image/jpeg',
        size: asset.fileSize || 0,
      };

      processSelectedFile(file);
    });
  }, [processSelectedFile]);

  /**
   * Handle file browser selection
   */
  const handleFileBrowse = useCallback(async () => {
    setError(null);

    try {
      console.log('[DocumentPickerModal] Opening file browser...');

      // Add more detailed logging
      console.log(
        '[DocumentPickerModal] DocumentPicker object:',
        DocumentPicker
      );
      console.log(
        '[DocumentPickerModal] Attempting to pick file with options:',
        {
          type: ['image/*', 'application/pdf'],
          allowMultiSelection: false,
        }
      );

      const results = await DocumentPicker.pick({
        type: ['image/*', 'application/pdf'],
        allowMultiSelection: false,
      }).catch(e => {
        // Add catch block to the promise itself
        console.error('[DocumentPickerModal] Pick promise rejected:', e);
        throw e;
      });

      console.log('[DocumentPickerModal] Pick succeeded, results:', results);

      const result = results[0];
      console.log('[DocumentPickerModal] First result:', result);

      const file: RNFile = {
        uri: result.uri,
        name: result.name || 'document',
        type: result.type || 'application/octet-stream',
        size: result.size || 0,
      };

      processSelectedFile(file);
    } catch (err) {
      // Type the error properly
      const error = err as {code?: string; message?: string};

      console.error('[DocumentPickerModal] Detailed error:', {
        error,
        code: error.code,
        message: error.message,
        stack: (error as Error).stack,
      });

      if (error.code === 'DOCUMENT_PICKER_CANCELED') {
        console.log('[DocumentPickerModal] File selection cancelled');
        return;
      }

      console.error('[DocumentPickerModal] File browser error:', error);
      setError('Failed to select file. Please try again.');
    }
  }, [processSelectedFile]);

  /**
   * Clear error message
   */
  const clearError = useCallback(() => {
    setError(null);
  }, []);

  return (
    <Portal>
      <Modal
        visible={visible}
        onDismiss={onClose}
        contentContainerStyle={styles.modalContainer}
        testID={testID}
      >
        <Card style={[styles.card, {backgroundColor: theme.colors.surface}]}>
          <Card.Content>
            <Text
              variant="headlineSmall"
              style={[styles.title, {color: theme.colors.onSurface}]}
              accessibilityRole="header"
              accessibilityLabel="Select document source"
            >
              Select Document
            </Text>

            <Text
              variant="bodyMedium"
              style={[styles.subtitle, {color: theme.colors.onSurfaceVariant}]}
            >
              Choose how you'd like to add your document
            </Text>

            {error && (
              <ErrorMessage
                message={error}
                type="error"
                visible={true}
                dismissible={true}
                onDismiss={clearError}
                style={styles.errorMessage}
                testID="document-picker-error"
              />
            )}

            <View style={styles.optionsContainer}>
              <Button
                mode="outlined"
                onPress={handleGalleryPick}
                disabled={isProcessing}
                style={styles.optionButton}
                contentStyle={styles.buttonContent}
                labelStyle={[styles.buttonLabel, {color: theme.colors.primary}]}
                icon={({size, color}) => (
                  <MaterialIcons
                    name="photo-library"
                    size={size}
                    color={color}
                  />
                )}
                accessibilityLabel="Select image from gallery"
                accessibilityRole="button"
                accessibilityHint="Opens your photo gallery to select an image"
                testID="gallery-button"
              >
                Gallery
              </Button>

              <Button
                mode="outlined"
                onPress={handleCameraCapture}
                disabled={isProcessing}
                style={styles.optionButton}
                contentStyle={styles.buttonContent}
                labelStyle={[styles.buttonLabel, {color: theme.colors.primary}]}
                icon={({size, color}) => (
                  <MaterialIcons name="camera-alt" size={size} color={color} />
                )}
                accessibilityLabel="Take photo with camera"
                accessibilityRole="button"
                accessibilityHint="Opens camera to take a new photo"
                testID="camera-button"
              >
                Camera
              </Button>

              <Button
                mode="outlined"
                onPress={handleFileBrowse}
                disabled={isProcessing}
                style={styles.optionButton}
                contentStyle={styles.buttonContent}
                labelStyle={[styles.buttonLabel, {color: theme.colors.primary}]}
                icon={({size, color}) => (
                  <MaterialIcons name="folder-open" size={size} color={color} />
                )}
                accessibilityLabel="Browse files"
                accessibilityRole="button"
                accessibilityHint="Opens file browser to select a document"
                testID="browse-button"
              >
                Browse Files
              </Button>
            </View>

            <Divider style={styles.divider} />

            <Text
              variant="bodySmall"
              style={[styles.helpText, {color: theme.colors.onSurfaceVariant}]}
            >
              Supported formats: JPG, PNG, PDF{'\n'}
              Maximum file size: 10MB
            </Text>
          </Card.Content>

          <Card.Actions style={styles.actions}>
            <Button
              mode="text"
              onPress={onClose}
              disabled={isProcessing}
              accessibilityLabel="Cancel document selection"
              accessibilityRole="button"
              testID="cancel-button"
            >
              Cancel
            </Button>
          </Card.Actions>
        </Card>
      </Modal>
    </Portal>
  );
};

const styles = StyleSheet.create({
  modalContainer: {
    margin: 20,
    justifyContent: 'center',
  },
  card: {
    borderRadius: 12,
    elevation: 8,
  },
  title: {
    textAlign: 'center',
    marginBottom: 8,
    fontWeight: 'bold',
  },
  subtitle: {
    textAlign: 'center',
    marginBottom: 24,
  },
  errorMessage: {
    marginBottom: 16,
  },
  optionsContainer: {
    gap: 12,
    marginBottom: 16,
  },
  optionButton: {
    borderRadius: 8,
    borderWidth: 1.5,
  },
  buttonContent: {
    paddingVertical: 8,
  },
  buttonLabel: {
    fontSize: 16,
    fontWeight: '500',
  },
  divider: {
    marginVertical: 16,
  },
  helpText: {
    textAlign: 'center',
    lineHeight: 18,
  },
  actions: {
    justifyContent: 'center',
    paddingTop: 8,
  },
});

export default DocumentPickerModal;
