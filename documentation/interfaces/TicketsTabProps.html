import {useNavigation, useFocusEffect} from '@react-navigation/native';
import React, {useCallback, useMemo, useState} from 'react';
import {
  View,
  Text,
  FlatList,
  StyleSheet,
  ActivityIndicator,
  RefreshControl,
} from 'react-native';
import {Button, FAB, Snackbar} from 'react-native-paper';
import MaterialIcons from 'react-native-vector-icons/MaterialIcons';
import {useGetTicketsQuery} from '../../../api/endpoints/tickets';
import CreateTicketModal from '../../../components/modals/CreateTicketModal';
import {TicketListItemComponent} from '../../../components/tickets';
import {useRefreshControl} from '../../../hooks/useRefreshControl';
import {useTicketPolling} from '../../../hooks/useTicketPolling';
import {useAppSelector} from '../../../hooks/useTypedRedux';
import type {RootStackScreenProps} from '../../../navigation/types';
import {selectTicketsByLeadId} from '../../../store/ticketsSlice';
import {useAppTheme} from '../../../theme/ThemeProvider';
import type {TicketListItem as TicketListItemType} from '../../../types/Ticket';

export interface TicketsTabProps {
  leadId: string;
}

/**
 * TicketsTab Component
 * Shows support tickets for a specific lead
 */
const TicketsTab: React.FC<TicketsTabProps> = ({leadId}) => {
  const theme = useAppTheme();
  const navigation =
    useNavigation<RootStackScreenProps<'TicketThread'>['navigation']>();

  // Modal state
  const [createTicketModalVisible, setCreateTicketModalVisible] =
    useState(false);
  const [snackbarVisible, setSnackbarVisible] = useState(false);
  const [snackbarMessage, setSnackbarMessage] = useState('');

  // Fetch all tickets (will be filtered by leadId)
  const {
    data: apiTickets,
    isLoading,
    isError,
    isFetching,
  } = useGetTicketsQuery(undefined, {
    selectFromResult: result => ({
      ...result,
      data: result.data || {items: [], total: 0, offset: 0, limit: 25},
    }),
  });

  // Get tickets filtered by leadId from Redux state
  const leadTickets = useAppSelector(selectTicketsByLeadId(leadId));

  // Pull-to-refresh functionality
  const {refreshTickets, canRefreshTickets} = useRefreshControl();

  // Ticket polling with focus-based lifecycle
  const ticketPolling = useTicketPolling({
    enabled: true,
    startImmediately: false, // Only start via focus control
  });

  /**
   * Handle tab focus - start polling
   */
  useFocusEffect(
    useCallback(() => {
      console.log(
        '[TicketsTab] Tab focused, starting ticket polling for lead:',
        leadId
      );
      ticketPolling.startPolling();

      return () => {
        console.log(
          '[TicketsTab] Tab unfocused, stopping ticket polling for lead:',
          leadId
        );
        ticketPolling.stopPolling();
      };
    }, [ticketPolling, leadId])
  );

  /**
   * Handle pull-to-refresh
   */
  const handleRefresh = useCallback(async () => {
    console.log('[TicketsTab] Pull to refresh triggered for lead:', leadId);
    try {
      await refreshTickets();
    } catch (error) {
      console.error('[TicketsTab] Refresh error:', error);
    }
  }, [refreshTickets, leadId]);

  /**
   * Handle ticket item press - navigate to ticket thread
   */
  const handleTicketPress = useCallback(
    (ticketId: string) => {
      console.log('[TicketsTab] Navigating to ticket thread:', ticketId);
      // Navigate to TicketThread screen (escape the tab navigator)
      (navigation as any).navigate('TicketThread', {ticketId});
    },
    [navigation]
  );

  /**
   * Handle create ticket button press
   */
  const handleCreateTicket = useCallback(() => {
    console.log('[TicketsTab] Opening create ticket modal for lead:', leadId);
    setCreateTicketModalVisible(true);
  }, [leadId]);

  /**
   * Handle close create ticket modal
   */
  const handleCloseCreateTicketModal = useCallback(() => {
    setCreateTicketModalVisible(false);
  }, []);

  /**
   * Show snackbar message
   */
  const showMessage = useCallback((message: string) => {
    setSnackbarMessage(message);
    setSnackbarVisible(true);
  }, []);

  /**
   * Handle ticket created successfully
   */
  const handleTicketCreated = useCallback(
    (ticketId: string) => {
      console.log('[TicketsTab] Ticket created successfully:', ticketId);
      showMessage(`Ticket ${ticketId} created successfully!`);
      // Optionally navigate to the new ticket
      // (navigation as any).navigate('TicketThread', {ticketId});
    },
    [showMessage]
  );

  /**
   * FlatList optimizations
   */
  const keyExtractor = useCallback(
    (item: TicketListItemType) => item.ticketId,
    []
  );

  const renderTicketItem = useCallback(
    ({item}: {item: TicketListItemType}) => (
      <TicketListItemComponent
        ticket={item}
        onPress={handleTicketPress}
        testID={`ticket-item-${item.ticketId}`}
      />
    ),
    [handleTicketPress]
  );

  const getItemLayout = useCallback(
    (_: any, index: number) => ({
      length: 100, // Approximate height of TicketListItemComponent
      offset: 100 * index,
      index,
    }),
    []
  );

  /**
   * Memoized empty state component
   */
  const EmptyState = useMemo(
    () => (
      <View style={styles.emptyContainer}>
        <MaterialIcons
          name="support-agent"
          size={64}
          color={theme.colors.onSurfaceVariant}
        />
        <Text style={[styles.emptyTitle, {color: theme.colors.onSurface}]}>
          No Support Tickets
        </Text>
        <Text
          style={[
            styles.emptyDescription,
            {color: theme.colors.onSurfaceVariant},
          ]}
        >
          No support tickets found for this lead. Create a new ticket to get
          help with your requests.
        </Text>
        <Button
          mode="contained"
          onPress={handleCreateTicket}
          style={styles.createButton}
          icon="plus"
        >
          Create Ticket
        </Button>
      </View>
    ),
    [theme, handleCreateTicket]
  );

  /**
   * Memoized error state component
   */
  const ErrorState = useMemo(
    () => (
      <View style={styles.errorContainer}>
        <MaterialIcons
          name="error-outline"
          size={64}
          color={theme.colors.error}
        />
        <Text style={[styles.errorTitle, {color: theme.colors.error}]}>
          Failed to Load Tickets
        </Text>
        <Text
          style={[
            styles.errorDescription,
            {color: theme.colors.onSurfaceVariant},
          ]}
        >
          Please pull down to refresh and try again.
        </Text>
        <Button
          mode="outlined"
          onPress={handleRefresh}
          style={styles.retryButton}
          icon="refresh"
        >
          Retry
        </Button>
      </View>
    ),
    [theme, handleRefresh]
  );

  // Show loading only if we're loading AND have no cached data for this lead
  const shouldShowLoading = isLoading && leadTickets.length === 0;

  // Loading state
  if (shouldShowLoading) {
    return (
      <View style={styles.loadingContainer}>
        <ActivityIndicator size="large" color={theme.colors.primary} />
        <Text
          style={[styles.loadingText, {color: theme.colors.onSurfaceVariant}]}
        >
          Loading tickets...
        </Text>
      </View>
    );
  }

  return (
    <View
      style={[styles.container, {backgroundColor: theme.colors.background}]}
    >
      <FlatList
        data={leadTickets} // Use filtered tickets for this lead
        renderItem={renderTicketItem}
        keyExtractor={keyExtractor}
        getItemLayout={getItemLayout}
        refreshControl={
          <RefreshControl
            refreshing={isFetching && canRefreshTickets()}
            onRefresh={handleRefresh}
            colors={[theme.colors.primary]}
            tintColor={theme.colors.primary}
          />
        }
        // Performance optimizations
        removeClippedSubviews={true}
        maxToRenderPerBatch={10}
        windowSize={10}
        initialNumToRender={10}
        // States
        ListEmptyComponent={isError ? ErrorState : EmptyState}
        // Styling
        contentContainerStyle={
          leadTickets.length === 0
            ? styles.emptyListContainer
            : styles.listContainer
        }
        showsVerticalScrollIndicator={false}
      />

      {/* Create Ticket FAB - Always visible, prefilled with leadId */}
      <FAB
        style={[styles.fab, {backgroundColor: theme.colors.primary}]}
        icon="plus"
        onPress={handleCreateTicket}
        label="Create Ticket"
        visible={true}
      />

      {/* Create Ticket Modal with prefilled leadId */}
      <CreateTicketModal
        visible={createTicketModalVisible}
        onClose={handleCloseCreateTicketModal}
        onSuccess={handleTicketCreated}
        prefilledLeadId={leadId} // This will disable the dropdown and prefill the lead
        testID="create-ticket-modal"
      />

      {/* Snackbar for success messages */}
      <Snackbar
        visible={snackbarVisible}
        onDismiss={() => {
          setSnackbarVisible(false);
          setSnackbarMessage('');
        }}
        duration={4000}
        action={{
          label: 'Dismiss',
          onPress: () => {
            setSnackbarVisible(false);
            setSnackbarMessage('');
          },
        }}
      >
        {snackbarMessage}
      </Snackbar>
    </View>
  );
};
