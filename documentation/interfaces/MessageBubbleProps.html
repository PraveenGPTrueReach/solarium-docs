import React from 'react';
import {View, Text} from 'react-native';
import {StyleSheet} from 'react-native';
import {spacing, borderRadius} from '../../theme/palette';
import {useAppTheme} from '../../theme/ThemeProvider';
import type {TicketMessage} from '../../types/Ticket';
import {getRelativeTime} from '../../utils/dateHelpers';

export interface MessageBubbleProps {
  /** Message data to display */
  message: TicketMessage;
  /** Test ID for testing */
  testID?: string;
}

/**
 * Get message bubble styling based on sender role
 */
const getMessageStyle = (sender: string, theme: any) => {
  const isCustomer = sender === 'Customer';

  return {
    bubble: {
      backgroundColor: isCustomer
        ? theme.colors.primary
        : theme.colors.surfaceVariant,
      alignSelf: isCustomer ? ('flex-end' as const) : ('flex-start' as const),
      marginLeft: isCustomer ? 48 : 0,
      marginRight: isCustomer ? 0 : 48,
    },
    text: {
      color: isCustomer
        ? theme.colors.onPrimary
        : theme.colors.onSurfaceVariant,
    },
    timestamp: {
      color: isCustomer
        ? `${theme.colors.onPrimary}CC` // 80% opacity
        : theme.colors.onSurfaceVariant,
    },
  };
};

/**
 * MessageBubble Component
 * Memoized for performance in FlatList
 */
const MessageBubble: React.FC<MessageBubbleProps> = React.memo(
  ({message, testID}) => {
    const theme = useAppTheme();
    const messageStyle = getMessageStyle(message.sender, theme);
    const relativeTime = getRelativeTime(message.timestamp);

    return (
      <View style={styles.container} testID={testID}>
        <View style={[styles.bubble, messageStyle.bubble]}>
          {/* Message Content */}
          <Text
            style={[styles.messageText, messageStyle.text]}
            accessible={true}
            accessibilityRole="text"
            accessibilityLabel={`Message from ${message.sender}: ${message.message}`}
          >
            {message.message}
          </Text>

          {/* Message Footer */}
          <Text
            style={[styles.timestamp, messageStyle.timestamp]}
            accessible={false}
          >
            {relativeTime}
          </Text>
        </View>
      </View>
    );
  }
);

MessageBubble.displayName = 'MessageBubble';

/**
 * MessageBubble Styles
 * Styling for message bubble component following chat UI patterns
 */
const styles = StyleSheet.create({
  container: {
    paddingHorizontal: spacing.md,
    paddingVertical: spacing.xs,
  },

  bubble: {
    borderRadius: borderRadius.medium,
    padding: spacing.md,
    maxWidth: '85%',
    elevation: 1,
    shadowColor: '#000',
    shadowOffset: {width: 0, height: 1},
    shadowOpacity: 0.1,
    shadowRadius: 2,
  },

  timestamp: {
    fontSize: 11,
    fontWeight: '400',
    alignSelf: 'flex-end',
    opacity: 0.8,
  },

  messageText: {
    fontSize: 14,
    lineHeight: 20,
    marginBottom: spacing.xs,
  },
});

export default MessageBubble;
