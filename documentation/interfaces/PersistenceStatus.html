import {useEffect, useState, useCallback} from 'react';
import {persistor} from '../store';

interface PersistenceStatus {
  /** Whether persistence is currently in progress */
  isPersisting: boolean;
  /** Latest persistence error, if any */
  error: Error | null;
  /** Manual flush function to trigger persistence */
  flush: () => Promise<void>;
}

export const usePersistenceStatus = (): PersistenceStatus => {
  const [isPersisting, setIsPersisting] = useState(false);
  const [error, setError] = useState<Error | null>(null);

  /**
   * Manual flush function
   */
  const flush = useCallback(async (): Promise<void> => {
    try {
      setIsPersisting(true);
      setError(null);

      console.log('[usePersistenceStatus] Starting manual flush...');

      // Trigger persistence flush
      await persistor.flush();

      console.log('[usePersistenceStatus] Manual flush completed');
    } catch (err) {
      const error =
        err instanceof Error ? err : new Error('Persistence failed');
      console.error('[usePersistenceStatus] Flush error:', error);
      setError(error);
    } finally {
      setIsPersisting(false);
    }
  }, []);

  /**
   * Monitor persistor state changes
   */
  useEffect(() => {
    const handlePersistorStateChange = () => {
      const state = persistor.getState();

      // Check if persistence is in progress
      const isCurrentlyPersisting = state.registry.length > 0;

      if (isCurrentlyPersisting !== isPersisting) {
        console.log('[usePersistenceStatus] Persistence state changed:', {
          isPersisting: isCurrentlyPersisting,
          registry: state.registry,
        });
        setIsPersisting(isCurrentlyPersisting);
      }
    };

    // Initial check
    handlePersistorStateChange();

    // Subscribe to changes
    const unsubscribe = persistor.subscribe(handlePersistorStateChange);

    return () => {
      unsubscribe();
    };
  }, [isPersisting]);

  return {
    isPersisting,
    error,
    flush,
  };
};

export default usePersistenceStatus;
