import {createSelector, createSlice, PayloadAction} from '@reduxjs/toolkit';
import type {Service} from '../types/Service';

export interface CartItem {
  id: string;
  qty: number;
  service: Service;
  addedAt: string;
}

export interface ServicesCartState {
  items: Record<string, CartItem>;
  totalItems: number;
  lastUpdated: string | null;
}

const initialState: ServicesCartState = {
  items: {},
  totalItems: 0,
  lastUpdated: null,
};

/**
 * Validate cart state for persistence recovery
 */
export const validateCartState = (state: any): boolean => {
  if (!state || typeof state !== 'object') {
    return false;
  }
  if (!state.items || typeof state.items !== 'object') {
    return false;
  }
  if (typeof state.totalItems !== 'number') {
    return false;
  }

  // Validate each cart item
  for (const [key, item] of Object.entries(state.items)) {
    if (!item || typeof item !== 'object') {
      return false;
    }
    const cartItem = item as any;
    if (!cartItem.id || typeof cartItem.id !== 'string') {
      return false;
    }
    if (typeof cartItem.qty !== 'number' || cartItem.qty < 1) {
      return false;
    }
    if (!cartItem.service || typeof cartItem.service !== 'object') {
      return false;
    }
    if (!cartItem.service.serviceId || cartItem.service.serviceId !== key) {
      return false;
    }
  }

  return true;
};

/**
 * Calculate total items in cart
 */
const calculateTotalItems = (items: Record<string, CartItem>): number => {
  return Object.values(items).reduce((total, item) => total + item.qty, 0);
};

/**
 * Check if adding item would exceed max limit
 */
const wouldExceedMaxItems = (
  items: Record<string, CartItem>,
  serviceId: string,
  qtyToAdd: number
): boolean => {
  const currentTotal = calculateTotalItems(items);
  const existingQty = items[serviceId]?.qty || 0;
  const newTotal = currentTotal - existingQty + (existingQty + qtyToAdd);
  return newTotal > 20;
};

export const servicesCartSlice = createSlice({
  name: 'servicesCart',
  initialState,
  reducers: {
    /**
     * Add service to cart or increase existing quantity
     *
     * Enforces 20-item total limit and validates input.
     * If service exists, increases quantity. If new, creates cart item.
     *
     * @param action.payload.service - Complete service object to add
     * @param action.payload.qty - Quantity to add (default: 1)
     *
     * @example
     * ```typescript
     * dispatch(addToCart({
     *   service: { serviceId: 'solar-install', name: 'Installation', ... },
     *   qty: 3
     * }));
     * ```
     */
    addToCart: (
      state,
      action: PayloadAction<{service: Service; qty?: number}>
    ) => {
      const {service, qty = 1} = action.payload;
      const serviceId = service.serviceId;

      // Check max items constraint
      if (wouldExceedMaxItems(state.items, serviceId, qty)) {
        console.warn(
          `[Cart] Cannot add ${qty} of ${serviceId}. Would exceed 20 item limit.`
        );
        return; // Don't modify state if it would exceed limit
      }

      const timestamp = new Date().toISOString();

      if (state.items[serviceId]) {
        // Update existing item
        state.items[serviceId].qty += qty;
        state.items[serviceId].service = service; // Update service data
      } else {
        // Add new item
        state.items[serviceId] = {
          id: serviceId,
          qty,
          service,
          addedAt: timestamp,
        };
      }

      state.totalItems = calculateTotalItems(state.items);
      state.lastUpdated = timestamp;
    },

    /**
     * Remove service from cart or decrease quantity
     *
     * If quantity to remove >= current quantity, removes service completely.
     * Otherwise, decreases quantity by specified amount.
     *
     * @param action.payload.serviceId - Service ID to remove from
     * @param action.payload.qty - Quantity to remove (default: 1)
     */
    removeFromCart: (
      state,
      action: PayloadAction<{serviceId: string; qty?: number}>
    ) => {
      const {serviceId, qty = 1} = action.payload;
      const existingItem = state.items[serviceId];

      if (!existingItem) {
        console.warn(`[Cart] Cannot remove ${serviceId}. Not in cart.`);
        return;
      }

      const timestamp = new Date().toISOString();

      if (existingItem.qty <= qty) {
        // Remove item completely
        delete state.items[serviceId];
      } else {
        // Decrease quantity
        state.items[serviceId].qty -= qty;
      }

      state.totalItems = calculateTotalItems(state.items);
      state.lastUpdated = timestamp;
    },

    /**
     * Set specific quantity for a service
     *
     * More efficient than multiple add/remove operations.
     * Setting quantity to 0 removes the service.
     * Validates 20-item total limit.
     *
     * @param action.payload.serviceId - Service ID to update
     * @param action.payload.qty - New quantity (0 removes service)
     */
    updateQuantity: (
      state,
      action: PayloadAction<{serviceId: string; qty: number}>
    ) => {
      const {serviceId, qty} = action.payload;
      const existingItem = state.items[serviceId];

      if (!existingItem) {
        console.warn(`[Cart] Cannot update ${serviceId}. Not in cart.`);
        return;
      }

      if (qty <= 0) {
        // Remove item if quantity is 0 or negative
        delete state.items[serviceId];
      } else {
        // Check max items constraint
        const tempItems = {...state.items};
        tempItems[serviceId] = {...existingItem, qty};

        if (calculateTotalItems(tempItems) > 20) {
          console.warn(
            `[Cart] Cannot set ${serviceId} quantity to ${qty}. Would exceed 20 item limit.`
          );
          return;
        }

        state.items[serviceId].qty = qty;
      }

      state.totalItems = calculateTotalItems(state.items);
      state.lastUpdated = new Date().toISOString();
    },

    /**
     * Clear entire cart
     */
    clearCart: state => {
      state.items = {};
      state.totalItems = 0;
      state.lastUpdated = new Date().toISOString();
    },

    /**
     * Remove specific service completely
     */
    removeService: (state, action: PayloadAction<{serviceId: string}>) => {
      const {serviceId} = action.payload;

      if (state.items[serviceId]) {
        delete state.items[serviceId];
        state.totalItems = calculateTotalItems(state.items);
        state.lastUpdated = new Date().toISOString();
      }
    },

    /**
     * Reset cart state (for testing/corruption recovery)
     */
    resetCartState: () => initialState,
  },
});

// Export actions
export const {
  addToCart,
  removeFromCart,
  updateQuantity,
  clearCart,
  removeService,
  resetCartState,
} = servicesCartSlice.actions;

// Selectors
export const selectCartItems = (state: {
  servicesCart: ServicesCartState | undefined;
}) => state.servicesCart?.items ?? {};

export const selectCartItemsArray = createSelector([selectCartItems], items =>
  Object.values(items)
);

export const selectCartCount = (state: {
  servicesCart: ServicesCartState | undefined;
}) => state.servicesCart?.totalItems ?? 0;

export const selectCartItemById =
  (serviceId: string) =>
  (state: {servicesCart: ServicesCartState | undefined}) =>
    state.servicesCart?.items[serviceId] || null;

export const selectCartIsEmpty = (state: {
  servicesCart: ServicesCartState | undefined;
}) => (state.servicesCart?.totalItems ?? 0) === 0;

export const selectCartLastUpdated = (state: {
  servicesCart: ServicesCartState | undefined;
}) => state.servicesCart?.lastUpdated ?? null;

export const selectUniqueItemsCount = (state: {
  servicesCart: ServicesCartState | undefined;
}) => Object.keys(state.servicesCart?.items ?? {}).length;

export const selectIsCartFull = (state: {
  servicesCart: ServicesCartState | undefined;
}) => (state.servicesCart?.totalItems ?? 0) >= 20;

// Export reducer
export default servicesCartSlice.reducer;
