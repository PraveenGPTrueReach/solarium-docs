import React from 'react';
import {View, Text, TouchableOpacity} from 'react-native';
import {Chip} from 'react-native-paper';
import MaterialIcons from 'react-native-vector-icons/MaterialIcons';
import {useAppTheme} from '../../../theme/ThemeProvider';
import type {Lead} from '../../../types/Lead';
import {getRelativeTime} from '../../../utils/dateHelpers';
import {styles} from './LeadListItem.styles';

export interface LeadListItemProps {
  /** Lead data to display */
  lead: Lead;
  /** Callback when item is pressed */
  onPress: (leadId: string) => void;
  /** Test ID for testing */
  testID?: string;
}

/**
 * Get status color and text color based on lead status
 * AC-4.2: Status colors follow palette mapping with proper contrast
 */
const getStatusConfig = (
  status: string,
  theme: any
): {backgroundColor: string; textColor: string} => {
  const statusLower = status.toLowerCase();

  console.log(
    '[LeadListItem] Status mapping for:',
    status,
    '-> lowercase:',
    statusLower
  );

  if (statusLower.includes('new')) {
    const config = {
      backgroundColor: theme.colors.info || '#2196F3',
      textColor: '#FFFFFF',
    };
    console.log('[LeadListItem] NEW status config:', config);
    return config;
  } else if (
    statusLower.includes('discussion') ||
    statusLower.includes('meeting')
  ) {
    // Use explicit purple since theme.colors.quoted might not exist
    const config = {backgroundColor: '#673AB7', textColor: '#FFFFFF'};
    console.log('[LeadListItem] DISCUSSION status config:', config);
    return config;
  } else if (
    statusLower.includes('customer') &&
    statusLower.includes('accepted')
  ) {
    const config = {
      backgroundColor: theme.colors.warning || '#FF9800',
      textColor: '#FFFFFF',
    };
    console.log('[LeadListItem] CUSTOMER_ACCEPTED status config:', config);
    return config;
  } else if (statusLower.includes('won') || statusLower.includes('executed')) {
    const config = {
      backgroundColor: theme.colors.success || '#4CAF50',
      textColor: '#FFFFFF',
    };
    console.log('[LeadListItem] WON status config:', config);
    return config;
  } else if (
    statusLower.includes('not interested') ||
    statusLower.includes('not responding')
  ) {
    const config = {
      backgroundColor: theme.colors.error || '#F44336',
      textColor: '#FFFFFF',
    };
    console.log('[LeadListItem] NOT_INTERESTED status config:', config);
    return config;
  } else {
    const config = {
      backgroundColor: theme.colors.primary || '#2E7D32',
      textColor: '#FFFFFF',
    };
    console.log('[LeadListItem] DEFAULT status config:', config);
    return config;
  }
};

/**
 * Split address into two lines for better display
 * AC-4.1: siteAddress (two-line)
 */
const splitAddress = (address: string): {line1: string; line2: string} => {
  if (!address) {
    return {line1: 'No address', line2: ''};
  }

  const parts = address.split(',').map(part => part.trim());

  if (parts.length <= 2) {
    return {line1: parts[0] || '', line2: parts[1] || ''};
  }

  // Split roughly in middle, but keep locality together
  const midPoint = Math.ceil(parts.length / 2);
  const line1 = parts.slice(0, midPoint).join(', ');
  const line2 = parts.slice(midPoint).join(', ');

  return {line1, line2};
};

/**
 * LeadListItem Component
 * Memoized for performance (AC: 50 fps on 6-year-old Android)
 */
const LeadListItem: React.FC<LeadListItemProps> = React.memo(
  ({lead, onPress, testID}) => {
    const theme = useAppTheme();
    const statusConfig = getStatusConfig(lead.status, theme);
    const {line1, line2} = splitAddress(lead.address);
    const relativeTime = getRelativeTime(lead.updatedAt);

    // AC-4.3: Accessible label combines address + status
    const accessibleLabel = `Lead ${lead.leadId}, ${lead.customerName}, ${line1} ${line2}, Status: ${lead.status}, Updated ${relativeTime}`;

    const handlePress = () => {
      onPress(lead.id || lead.leadId); // Handle both id formats
    };

    return (
      <TouchableOpacity
        style={[styles.container, {backgroundColor: theme.colors.surface}]}
        onPress={handlePress}
        testID={testID}
        // AC-4.3: accessibilityRole='button'
        accessible={true}
        accessibilityRole="button"
        accessibilityLabel={accessibleLabel}
        accessibilityHint="Tap to view lead details"
      >
        {/* Lead Header - ID and Status */}
        <View style={styles.leadHeader}>
          <Text
            style={[styles.leadId, {color: theme.colors.primary}]}
            accessible={false} // Included in parent's accessibilityLabel
          >
            {lead.leadId}
          </Text>

          {/* AC-4.3: Status Chip exposes accessible label */}
          <Chip
            style={[
              styles.statusChip,
              {backgroundColor: statusConfig.backgroundColor},
            ]}
            textStyle={[styles.statusChipText, {color: statusConfig.textColor}]}
            compact={true}
            accessible={true}
            accessibilityRole="text"
            accessibilityLabel={`Status: ${lead.status}`}
            mode="flat"
          >
            {lead.status}
          </Chip>
        </View>

        {/* AC-4.1: siteAddress (two-line) */}
        <View style={styles.addressContainer}>
          <Text
            style={[styles.addressLine1, {color: theme.colors.onSurface}]}
            numberOfLines={1}
            accessible={false} // Included in parent's accessibilityLabel
          >
            {line1}
          </Text>
          {line2 ? (
            <Text
              style={[
                styles.addressLine2,
                {color: theme.colors.onSurfaceVariant},
              ]}
              numberOfLines={1}
              accessible={false} // Included in parent's accessibilityLabel
            >
              {line2}
            </Text>
          ) : null}
        </View>

        {/* Footer - Customer and Time */}
        <View style={styles.footer}>
          <View style={styles.customerInfo}>
            <MaterialIcons
              name="person"
              size={14}
              color={theme.colors.onSurfaceVariant}
              style={styles.customerIcon}
            />
            <Text
              style={[
                styles.customerName,
                {color: theme.colors.onSurfaceVariant},
              ]}
              numberOfLines={1}
              accessible={false} // Included in parent's accessibilityLabel
            >
              {lead.customerName}
            </Text>
          </View>

          {/* AC-4.1: updatedAt (relative) */}
          <View style={styles.timeContainer}>
            <MaterialIcons
              name="schedule"
              size={12}
              color={theme.colors.onSurfaceVariant}
              style={styles.timeIcon}
            />
            <Text
              style={[styles.timeText, {color: theme.colors.onSurfaceVariant}]}
              accessible={false} // Included in parent's accessibilityLabel
            >
              {relativeTime}
            </Text>
          </View>
        </View>
      </TouchableOpacity>
    );
  }
);

LeadListItem.displayName = 'LeadListItem';

export default LeadListItem;
