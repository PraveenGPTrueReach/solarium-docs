import React, {useState, useCallback} from 'react';
import {View, StyleSheet} from 'react-native';
import {Text, Button, ProgressBar, Chip} from 'react-native-paper';
import MaterialIcons from 'react-native-vector-icons/MaterialIcons';
import type {KycDocView} from '../../hooks/useKycDocuments';
import {useKycUpload} from '../../hooks/useKycUpload';
import {spacing, colors} from '../../theme/palette';
import {useAppTheme} from '../../theme/ThemeProvider';
import type {RNFile} from '../../types/File';
import type {KycDocumentStatus} from '../../types/KycDocument';
import {DocumentPickerModal} from '../forms';

/**
 * Document Row Props
 */
export interface DocumentRowProps {
  document: KycDocView;
  onUploadSuccess?: () => void;
  onUploadError?: (error: Error) => void;
  testID?: string;
}

/**
 * Get status badge color based on document status
 */
const getStatusColor = (status: KycDocumentStatus | 'Missing') => {
  switch (status) {
    case 'Approved':
      return colors.success; // Green
    case 'Pending':
      return colors.text.secondary; // Grey
    case 'Rejected':
      return colors.error; // Red
    case 'Missing':
    default:
      return colors.text.disabled; // Light grey
  }
};

/**
 * Get status text color for better contrast
 */
const getStatusTextColor = (status: KycDocumentStatus | 'Missing') => {
  switch (status) {
    case 'Approved':
    case 'Rejected':
      return '#FFFFFF'; // White text on colored background
    case 'Pending':
      return colors.text.primary; // Dark text
    case 'Missing':
    default:
      return colors.text.secondary; // Medium text
  }
};

/**
 * Get status icon
 */
const getStatusIcon = (status: KycDocumentStatus | 'Missing') => {
  switch (status) {
    case 'Approved':
      return 'check-circle';
    case 'Pending':
      return 'schedule';
    case 'Rejected':
      return 'cancel';
    case 'Missing':
    default:
      return 'radio-button-unchecked';
  }
};

/**
 * Document Row Component
 * Displays individual KYC document with status and upload functionality
 */
const DocumentRow: React.FC<DocumentRowProps> = ({
  document,
  onUploadSuccess,
  onUploadError,
  testID = 'document-row',
}) => {
  const theme = useAppTheme();
  const [showPicker, setShowPicker] = useState(false);

  // Upload hook
  const {
    startUpload,
    progress,
    status: uploadStatus,
    error,
    cancel,
  } = useKycUpload();

  /**
   * Handle file selection from picker
   */
  const handleFilePicked = useCallback(
    (file: RNFile) => {
      console.log(`[DocumentRow] File picked for ${document.docType}:`, file);
      startUpload(document.docType, file);
    },
    [document.docType, startUpload]
  );

  /**
   * Handle upload button press
   */
  const handleUploadPress = useCallback(() => {
    if (!document.canUpload) {
      console.warn(`[DocumentRow] Upload not allowed for ${document.docType}`);
      return;
    }
    setShowPicker(true);
  }, [document.canUpload, document.docType]);

  /**
   * Handle upload success
   */
  React.useEffect(() => {
    if (uploadStatus === 'success') {
      console.log(`[DocumentRow] Upload successful for ${document.docType}`);
      onUploadSuccess?.();
    }
  }, [uploadStatus, document.docType, onUploadSuccess]);

  /**
   * Handle upload error
   */
  React.useEffect(() => {
    if (uploadStatus === 'error' && error) {
      console.error(
        `[DocumentRow] Upload failed for ${document.docType}:`,
        error
      );
      onUploadError?.(error);
    }
  }, [uploadStatus, error, document.docType, onUploadError]);

  const isUploading = uploadStatus === 'running';
  const showUploadButton = document.canUpload && !isUploading;
  const statusColor = getStatusColor(document.status);
  const statusTextColor = getStatusTextColor(document.status);
  const statusIcon = getStatusIcon(document.status);

  return (
    <View
      style={[styles.container, {backgroundColor: theme.colors.surface}]}
      testID={`${testID}-${document.docType}`}
    >
      {/* Document Info Section */}
      <View style={styles.infoSection}>
        <View style={styles.nameSection}>
          <Text
            variant="titleMedium"
            style={[styles.documentName, {color: theme.colors.onSurface}]}
            testID={`${testID}-name`}
          >
            {document.displayName}
          </Text>

          {/* Status Badge */}
          <Chip
            icon={({size, color}) => (
              <MaterialIcons name={statusIcon} size={size} color={color} />
            )}
            style={[styles.statusChip, {backgroundColor: statusColor}]}
            textStyle={[styles.statusText, {color: statusTextColor}]}
            compact
            testID={`${testID}-status`}
            accessibilityLabel={`Status: ${document.status}`}
          >
            {document.status}
          </Chip>
        </View>

        {/* Upload Date (if document exists) */}
        {document.latestDoc && (
          <Text
            variant="bodySmall"
            style={[styles.uploadDate, {color: theme.colors.onSurfaceVariant}]}
            testID={`${testID}-date`}
          >
            Uploaded:{' '}
            {new Date(document.latestDoc.uploadedAt).toLocaleDateString()}
          </Text>
        )}
      </View>

      {/* Action Section */}
      <View style={styles.actionSection}>
        {/* Progress Bar (during upload) */}
        {isUploading && progress !== null && (
          <View style={styles.progressSection}>
            <ProgressBar
              progress={progress / 100}
              color={theme.colors.primary}
              style={styles.progressBar}
              accessibilityLabel={`Upload progress: ${progress}%`}
              testID={`${testID}-progress`}
            />
            <View style={styles.progressActions}>
              <Text
                variant="bodySmall"
                style={[
                  styles.progressText,
                  {color: theme.colors.onSurfaceVariant},
                ]}
              >
                {progress}%
              </Text>
              <Button
                mode="text"
                onPress={cancel}
                compact
                textColor={theme.colors.error}
                testID={`${testID}-cancel`}
                accessibilityLabel="Cancel upload"
              >
                Cancel
              </Button>
            </View>
          </View>
        )}

        {/* Upload Button */}
        {showUploadButton && (
          <Button
            mode="outlined"
            onPress={handleUploadPress}
            icon={({size, color}) => (
              <MaterialIcons name="cloud-upload" size={size} color={color} />
            )}
            style={styles.uploadButton}
            contentStyle={styles.uploadButtonContent}
            labelStyle={styles.uploadButtonLabel}
            testID={`${testID}-upload`}
            accessibilityRole="button"
            accessibilityLabel={`Upload ${document.displayName}`}
            accessibilityHint={`Opens file picker to upload ${document.displayName} document`}
          >
            Upload
          </Button>
        )}

        {/* Status Message for Non-Uploadable Documents */}
        {!document.canUpload && !isUploading && (
          <Text
            variant="bodySmall"
            style={[
              styles.statusMessage,
              {color: theme.colors.onSurfaceVariant},
            ]}
            testID={`${testID}-message`}
          >
            {document.status === 'Approved'
              ? 'Document approved'
              : 'Upload not available'}
          </Text>
        )}
      </View>

      {/* Document Picker Modal */}
      <DocumentPickerModal
        visible={showPicker}
        onClose={() => setShowPicker(false)}
        onFilePicked={handleFilePicked}
        testID={`${testID}-picker`}
      />
    </View>
  );
};
