import {createApi} from '@reduxjs/toolkit/query/react';
import type {
  GetTicketByIdResponse,
  GetTicketsResponse,
  TicketsQueryParams,
  CreateTicketRequest,
  CreateTicketResponse,
  AddMessageRequest,
} from '../../types/Ticket';
import {baseQuery, transformResponse} from '../baseQuery';

export interface ReopenTicketRequest {
  /** Reason for reopening (10-500 chars) */
  reason: string;
}

/**
 * Transform ticket API errors to consistent ServiceFetchError format
 * Follows the same pattern as notifications error transformation
 */
const transformTicketError = (error: any) => {
  // Handle HTTP errors with status code and backend message
  if (error.status && error.data) {
    const statusCode = error.status;
    const errorMessage = error.data.error || 'Request failed';

    return {
      name: 'ServiceFetchError',
      message: errorMessage,
      code: 'API_ERROR',
      operation: 'API_ERROR',
      httpStatus: statusCode,
      serviceId: undefined,
      originalError: undefined,
    };
  }

  // Handle network errors (no HTTP status)
  if (error.status === 'FETCH_ERROR') {
    return {
      name: 'ServiceFetchError',
      message: 'Network error - please check your connection',
      code: 'NETWORK_ERROR',
      operation: 'NETWORK_ERROR',
      httpStatus: 0,
      serviceId: undefined,
      originalError: undefined,
    };
  }

  // Handle timeout errors
  if (error.status === 'TIMEOUT_ERROR') {
    return {
      name: 'ServiceFetchError',
      message: 'Request timeout - please try again',
      code: 'TIMEOUT_ERROR',
      operation: 'TIMEOUT_ERROR',
      httpStatus: 408,
      serviceId: undefined,
      originalError: undefined,
    };
  }

  // Fallback for unexpected error formats
  return {
    name: 'ServiceFetchError',
    message: 'Unexpected error occurred',
    code: 'UNKNOWN_ERROR',
    operation: 'UNKNOWN_ERROR',
    httpStatus: 500,
    serviceId: undefined,
    originalError: undefined,
  };
};

/**
 * Tickets API definition using RTK Query
 * Provides endpoints for CRUD operations on support tickets
 */
export const ticketsApi = createApi({
  reducerPath: 'ticketsApi',
  baseQuery,
  tagTypes: ['Ticket', 'TicketsList'],
  endpoints: builder => ({
    /**
     * Get paginated tickets list for current user
     * Uses JWT token from auth state to identify user
     *
     * @param params - Optional query parameters for filtering/pagination
     */
    getTickets: builder.query<
      GetTicketsResponse,
      TicketsQueryParams | undefined
    >({
      query: (params = {}) => {
        const searchParams = new URLSearchParams();

        // Add pagination parameters with defaults
        searchParams.append('offset', (params.offset || 0).toString());
        searchParams.append('limit', (params.limit || 25).toString());

        // Add optional filter parameters
        if (params.status) {
          searchParams.append('status', params.status);
        }
        if (params.cpId) {
          searchParams.append('cpId', params.cpId);
        }
        if (params.customerId) {
          searchParams.append('customerId', params.customerId);
        }
        if (params.dateRange) {
          searchParams.append('dateRange', params.dateRange);
        }

        return `tickets?${searchParams.toString()}`;
      },
      transformResponse: transformResponse<GetTicketsResponse>,
      transformErrorResponse: transformTicketError,
      providesTags: result => {
        if (!result || !result.items) {
          return [{type: 'TicketsList' as const, id: 'LIST'}];
        }

        return [
          ...result.items.map(ticket => ({
            type: 'Ticket' as const,
            id: ticket.ticketId,
          })),
          {type: 'TicketsList' as const, id: 'LIST'},
        ];
      },
    }),

    /**
     * Get single ticket with full message thread
     * Used for detailed ticket view with all messages
     */
    getTicketById: builder.query<GetTicketByIdResponse, string>({
      query: ticketId => `tickets/${ticketId}`,
      transformResponse: transformResponse<GetTicketByIdResponse>,
      transformErrorResponse: transformTicketError,
      providesTags: (result, error, ticketId) => [
        {type: 'Ticket' as const, id: ticketId}, // Refresh Ticket
      ],
    }),

    /**
     * Create a new support ticket
     * Only works for leads with status "Executed"
     */
    createTicket: builder.mutation<CreateTicketResponse, CreateTicketRequest>({
      query: ticketData => ({
        url: 'tickets',
        method: 'POST',
        body: ticketData,
      }),
      transformResponse: transformResponse<CreateTicketResponse>,
      transformErrorResponse: transformTicketError,
      invalidatesTags: [{type: 'TicketsList' as const, id: 'LIST'}], // Refresh list
    }),

    /**
     * Add a message to an existing ticket thread
     */
    addMessage: builder.mutation<
      void,
      {ticketId: string; message: AddMessageRequest}
    >({
      query: ({ticketId, message}) => ({
        url: `tickets/${ticketId}/messages`,
        method: 'POST',
        body: message,
      }),
      transformResponse: transformResponse<void>,
      transformErrorResponse: transformTicketError,
      invalidatesTags: (result, error, {ticketId}) => [
        {type: 'Ticket' as const, id: ticketId}, // Refresh Ticket
        {type: 'TicketsList' as const, id: 'LIST'}, // Refresh list (for message count, etc.)
      ],
    }),

    /**
     * Reopen a closed ticket
     * Only works for closed tickets
     */
    reopenTicket: builder.mutation<
      void,
      {ticketId: string; reason: ReopenTicketRequest}
    >({
      query: ({ticketId, reason}) => ({
        url: `tickets/${ticketId}/reopen`,
        method: 'PATCH',
        body: reason,
      }),
      transformResponse: transformResponse<void>,
      transformErrorResponse: transformTicketError,
      invalidatesTags: (result, error, {ticketId}) => [
        {type: 'Ticket' as const, id: ticketId},
        {type: 'TicketsList' as const, id: 'LIST'},
      ],
    }),
  }),
});

// Export hooks for use in components
export const {
  useGetTicketsQuery,
  useGetTicketByIdQuery,
  useCreateTicketMutation,
  useAddMessageMutation,
  useReopenTicketMutation,
} = ticketsApi;

// Export the API for store configuration
export default ticketsApi;
