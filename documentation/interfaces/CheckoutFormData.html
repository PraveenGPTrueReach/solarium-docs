import {useFocusEffect, useNavigation} from '@react-navigation/native';
import {StackNavigationProp} from '@react-navigation/stack';
import React, {useCallback, useState} from 'react';
import {
  View,
  Text,
  FlatList,
  StyleSheet,
  Alert,
  ScrollView,
} from 'react-native';
import {Snackbar} from 'react-native-paper';
import MaterialIcons from 'react-native-vector-icons/MaterialIcons';
import {useCreateLeadMutation} from '../../api/endpoints/leads';
import {
  SafeAreaContainer,
  Button,
  TextInput,
  FormProviderWrapper,
  CartItemRow,
  Controller,
  LoadingOverlayContainer,
} from '../../components';
import {ServiceFetchError} from '../../errors';
import {useAppSelector, useAppDispatch} from '../../hooks/useTypedRedux';
import {useNewAuth} from '../../hooks/useNewAuth';
import type {RootStackParamList} from '../../navigation/types';
import {
  selectCartItemsArray,
  selectCartCount,
  selectCartIsEmpty,
  type CartItem,
  clearCart,
} from '../../store/servicesCartSlice';
import {useAppTheme} from '../../theme/ThemeProvider';
import type {CreateLeadRequest} from '../../types/Lead';

type CartCheckoutScreenNavigationProp = StackNavigationProp<
  RootStackParamList,
  'CartCheckout'
>;

/**
 * Form data interface (simplified to match API)
 */
interface CheckoutFormData {
  address: string;
}

/**
 * Cart Checkout Screen Component
 */
const CartCheckoutScreen: React.FC = () => {
  const theme = useAppTheme();
  const navigation = useNavigation<CartCheckoutScreenNavigationProp>();
  const dispatch = useAppDispatch();

  // API mutation hook
  const [createLead, {isLoading: isCreatingLead}] = useCreateLeadMutation();

  // Auth state from new auth system
  const {user: currentUser} = useNewAuth();

  // Local state
  const [snackbarVisible, setSnackbarVisible] = useState(false);
  const [snackbarMessage, setSnackbarMessage] = useState('');

  // Cart selectors
  const cartItems = useAppSelector(selectCartItemsArray);
  const cartCount = useAppSelector(selectCartCount);
  const isCartEmpty = useAppSelector(selectCartIsEmpty);

  // Form default values
  const defaultValues: CheckoutFormData = {
    address: '',
  };

  /**
   * Handle empty cart redirect
   */
  const handleEmptyCartRedirect = useCallback(() => {
    if (isCartEmpty) {
      Alert.alert(
        'Empty Cart',
        'Your cart is empty. Please add services before checkout.',
        [
          {
            text: 'OK',
            onPress: () => navigation.goBack(),
          },
        ]
      );
    }
  }, [isCartEmpty, navigation]);

  /**
   * Check for empty cart on screen focus
   */
  useFocusEffect(
    useCallback(() => {
      handleEmptyCartRedirect();
    }, [handleEmptyCartRedirect])
  );

  /**
   * Reset error state on screen focus
   */
  useFocusEffect(
    useCallback(() => {
      if (snackbarVisible) {
        setSnackbarVisible(false);
        setSnackbarMessage('');
      }
    }, [snackbarVisible])
  );

  /**
   * Show error snackbar
   */
  const showError = useCallback((message: string) => {
    setSnackbarMessage(message);
    setSnackbarVisible(true);
  }, []);

  /**
   * Show success snackbar
   */
  const showSuccess = useCallback((message: string) => {
    setSnackbarMessage(message);
    setSnackbarVisible(true);
  }, []);

  /**
   * Handle form submission with real API call
   */
  const handleSubmit = useCallback(
    async (data: CheckoutFormData) => {
      // Early guard - prevent double submissions
      if (isCreatingLead) {
        console.log('[CartCheckout] Submission already in progress, ignoring');
        return;
      }

      // Show confirmation alert before submitting
      Alert.alert(
        'Submit Service Request',
        `Are you sure you want to submit this service request?\n\nCustomer: ${
          currentUser?.name || 'Customer'
        }\nPhone: ${currentUser?.phone || 'N/A'}\nAddress: ${
          data.address
        }\nServices: ${cartCount} items\n\nThis will create a lead for our team to contact you.`,
        [
          {
            text: 'Cancel',
            style: 'cancel',
          },
          {
            text: 'Submit Request',
            style: 'default',
            onPress: async () => {
              try {
                console.log('[CartCheckout] Starting lead creation...', {
                  customerName: currentUser?.name || 'Customer',
                  phone: currentUser?.phone || 'N/A',
                  cartItems: cartItems.length,
                  totalItems: cartCount,
                });

                // Prepare API payload (matches swagger spec exactly)
                const serviceIds = cartItems.map(
                  item => item.service.serviceId
                );
                const createLeadPayload: CreateLeadRequest = {
                  customerName: currentUser?.name || `Customer_${Date.now()}`,
                  phone: currentUser?.phone || '0000000000', // Use actual phone or fallback
                  address: data.address.trim(),
                  services: serviceIds,
                };

                console.log(
                  '[CartCheckout] API payload (swagger compliant):',
                  createLeadPayload
                );

                // Call the API
                const result = await createLead(createLeadPayload).unwrap();

                // Handle the actual API response structure
                const leadId =
                  result.leadId || result.data?.leadId || 'Unknown';

                console.log('[CartCheckout] Lead created successfully:', {
                  leadId,
                  fullResponse: result,
                });

                // Clear cart immediately
                dispatch(clearCart());

                // Navigate to Services screen with success message
                navigation.navigate('Main', {
                  screen: 'Services',
                  params: {
                    showSuccessMessage: true,
                    successMessage: `Service request submitted successfully! Lead ID: ${leadId}`,
                    leadId,
                  },
                });
              } catch (error) {
                console.error('[CartCheckout] Lead creation failed:', error);

                // Handle different error types
                let errorMessage =
                  'Failed to submit service request. Please try again.';

                if (error instanceof ServiceFetchError) {
                  errorMessage = error.message;
                } else if (
                  error &&
                  typeof error === 'object' &&
                  'message' in error
                ) {
                  errorMessage = (error as any).message;
                } else if (
                  error &&
                  typeof error === 'object' &&
                  'data' in error
                ) {
                  const errorData = (error as any).data;
                  if (errorData && typeof errorData.message === 'string') {
                    errorMessage = errorData.message;
                  }
                }

                showError(errorMessage);
              }
            },
          },
        ]
      );
    },
    [
      cartItems,
      cartCount,
      currentUser,
      isCreatingLead,
      createLead,
      dispatch,
      navigation,
      showSuccess,
      showError,
    ]
  );

  /**
   * Calculate cart summary
   */
  const cartSummary = {
    totalItems: cartCount,
    uniqueServices: cartItems.length,
  };

  /**
   * Handle clear cart with confirmation
   */
  const handleClearCart = useCallback(() => {
    Alert.alert(
      'Clear Cart',
      'Are you sure you want to remove all items from your cart? This action cannot be undone.',
      [
        {
          text: 'Cancel',
          style: 'cancel',
        },
        {
          text: 'Clear Cart',
          style: 'destructive',
          onPress: () => {
            dispatch(clearCart());
            navigation.navigate('Main', {screen: 'Services'});
          },
        },
      ]
    );
  }, [dispatch, navigation]);

  /**
   * Render cart item
   */
  const renderCartItem = useCallback(
    ({item, index}: {item: CartItem; index: number}) => (
      <CartItemRow cartItem={item} testID={`cart-item-${index}`} />
    ),
    []
  );

  /**
   * Render cart summary header
   */
  const renderCartSummary = () => (
    <View style={styles.summaryContainer}>
      <View style={styles.summaryHeader}>
        <MaterialIcons
          name="shopping-cart"
          size={20}
          color={theme.colors.primary}
        />
        <Text style={[styles.summaryTitle, {color: theme.colors.onSurface}]}>
          Cart Summary
        </Text>
      </View>
      <View style={styles.summaryStats}>
        <Text
          style={[styles.summaryText, {color: theme.colors.onSurfaceVariant}]}
          accessible={true}
          accessibilityRole="text"
          accessibilityLabel={`${cartSummary.uniqueServices} unique services, ${cartSummary.totalItems} total items`}
        >
          {cartSummary.uniqueServices} Service
          {cartSummary.uniqueServices !== 1 ? 's' : ''} â€¢{' '}
          {cartSummary.totalItems} Item{cartSummary.totalItems !== 1 ? 's' : ''}
        </Text>
      </View>
    </View>
  );

  /**
   * Render customer info
   */
  const renderCustomerInfo = () => (
    <View style={styles.customerContainer}>
      <View style={styles.customerHeader}>
        <MaterialIcons name="person" size={20} color={theme.colors.primary} />
        <Text style={[styles.customerTitle, {color: theme.colors.onSurface}]}>
          Customer Information
        </Text>
      </View>
      <Text
        style={[styles.customerText, {color: theme.colors.onSurfaceVariant}]}
        accessible={true}
        accessibilityRole="text"
        accessibilityLabel={`Customer: ${
          currentUser?.name || 'Customer'
        }, Phone: ${currentUser?.phone || 'N/A'}`}
      >
        Name: {currentUser?.name || 'Customer'}
      </Text>
      <Text
        style={[styles.customerText, {color: theme.colors.onSurfaceVariant}]}
      >
        Phone: {currentUser?.phone || 'N/A'}
      </Text>
      {currentUser?.email && (
        <Text
          style={[styles.customerText, {color: theme.colors.onSurfaceVariant}]}
        >
          Email: {currentUser.email}
        </Text>
      )}
    </View>
  );

  /**
   * Render pricing footer
   */
  const renderPricingFooter = () => (
    <View
      style={[
        styles.pricingFooter,
        {backgroundColor: theme.colors.primaryContainer},
      ]}
    >
      <View style={styles.pricingRow}>
        <MaterialIcons
          name="info-outline"
          size={16}
          color={theme.colors.onPrimaryContainer}
        />
        <Text
          style={[styles.pricingText, {color: theme.colors.onPrimaryContainer}]}
          accessible={true}
          accessibilityRole="text"
          accessibilityLabel="Pricing information: Contact us for detailed quotation"
        >
          Contact us for detailed quotation based on your requirements
        </Text>
      </View>
    </View>
  );

  /**
   * Render form section (simplified - only address needed)
   */
  const renderFormSection = () => (
    <FormProviderWrapper
      defaultValues={defaultValues}
      formOptions={{
        mode: 'onChange',
        resolver: undefined,
      }}
      onSubmit={handleSubmit}
      testID="checkout-form"
    >
      {methods => (
        <View style={styles.formContainer}>
          <Text style={[styles.sectionTitle, {color: theme.colors.onSurface}]}>
            Service Details
          </Text>

          {/* Address Field - Only field needed */}
          <Controller
            name="address"
            control={methods.control}
            rules={{
              required: 'Address is required',
              minLength: {
                value: 10,
                message: 'Address must be at least 10 characters long',
              },
              validate: value => {
                const trimmed = value?.trim();
                if (!trimmed) {
                  return 'Address is required';
                }
                if (trimmed.length < 10) {
                  return 'Address must be at least 10 characters long';
                }
                return true;
              },
            }}
            render={({field, fieldState}) => (
              <TextInput
                label="Service Address *"
                placeholder="Enter the complete address where service is required"
                value={field.value}
                onChangeText={text => field.onChange(text)}
                errorMessage={fieldState.error?.message}
                hasError={!!fieldState.error}
                multiline
                numberOfLines={4}
                maxLength={300}
                testID="address-input"
                accessibilityLabel="Service address input field"
                accessibilityHint="Enter the complete address where solar service is required"
              />
            )}
          />

          {/* Info Text */}
          <Text
            style={[styles.infoText, {color: theme.colors.onSurfaceVariant}]}
          >
            Please provide a complete address including area, city, and any
            landmarks for easy location.
          </Text>

          {/* Submit Button */}
          <Button
            title={isCreatingLead ? 'Submitting...' : 'Submit Service Request'}
            onPress={methods.onSubmit}
            variant="primary"
            size="large"
            fullWidth
            disabled={
              isCreatingLead ||
              !methods.formState.isValid ||
              !!methods.formState.errors.address ||
              isCartEmpty
            }
            style={styles.submitButton}
            testID="submit-button"
            accessibilityLabel={
              isCreatingLead
                ? 'Submitting service request'
                : 'Submit service request'
            }
            accessibilityHint={
              isCreatingLead
                ? 'Please wait, submitting your request'
                : !methods.formState.isValid
                ? 'Form has errors, please correct them first'
                : isCartEmpty
                ? 'Cart is empty, cannot submit'
                : 'Submit your service request with selected services'
            }
          />
        </View>
      )}
    </FormProviderWrapper>
  );

  // Early return for empty cart (backup check)
  if (isCartEmpty) {
    return (
      <SafeAreaContainer testID="empty-cart-screen">
        <View style={styles.emptyContainer}>
          <MaterialIcons
            name="shopping-cart"
            size={64}
            color={theme.colors.onSurfaceVariant}
          />
          <Text style={[styles.emptyTitle, {color: theme.colors.onSurface}]}>
            Your cart is empty
          </Text>
          <Text
            style={[
              styles.emptyMessage,
              {color: theme.colors.onSurfaceVariant},
            ]}
          >
            Add some services to continue with checkout
          </Text>
        </View>
      </SafeAreaContainer>
    );
  }

  return (
    <SafeAreaContainer testID="cart-checkout-screen">
      {/* Loading Overlay */}
      {isCreatingLead && (
        <LoadingOverlayContainer
          message="Submitting your service request..."
          visible={true}
        />
      )}

      <ScrollView
        style={[styles.container, {backgroundColor: theme.colors.background}]}
        contentContainerStyle={styles.scrollContent}
        showsVerticalScrollIndicator={false}
        scrollEnabled={!isCreatingLead}
      >
        {/* Customer Info */}
        {renderCustomerInfo()}

        {/* Cart Summary */}
        {renderCartSummary()}

        {/* Cart Items List */}
        <View style={styles.cartItemsContainer}>
          <FlatList
            data={cartItems}
            keyExtractor={item => item.id}
            renderItem={renderCartItem}
            scrollEnabled={false}
            showsVerticalScrollIndicator={false}
          />
        </View>

        {/* Pricing Footer */}
        {renderPricingFooter()}

        {/* Clear Cart Button */}
        <View style={styles.clearCartContainer}>
          <Button
            title="Clear Cart"
            onPress={handleClearCart}
            variant="outline"
            size="small"
            disabled={isCreatingLead}
            style={styles.clearCartButton}
            textStyle={styles.clearCartButtonText}
            testID="clear-cart-button"
            accessibilityLabel="Clear all items from cart"
            accessibilityHint="Removes all services from your cart"
          />
        </View>

        {/* Form Section */}
        {renderFormSection()}
      </ScrollView>

      {/* Snackbar for success/error messages */}
      <Snackbar
        visible={snackbarVisible}
        onDismiss={() => {
          setSnackbarVisible(false);
          setSnackbarMessage('');
        }}
        duration={4000}
        action={{
          label: 'Dismiss',
          onPress: () => {
            setSnackbarVisible(false);
            setSnackbarMessage('');
          },
        }}
        style={styles.snackbar}
      >
        {snackbarMessage}
      </Snackbar>
    </SafeAreaContainer>
  );
};
