export interface OTPInputProps {
  /** Current OTP value */
  value: string;

  /** Callback when OTP value changes */
  onChangeText: (otp: string) => void;

  /** Callback when OTP input is completed (all digits entered) */
  onComplete?: (otp: string) => void;

  /** Number of OTP digits (default: 6) */
  length?: number;

  /** Whether the input is disabled */
  disabled?: boolean;

  /** Whether to show error state */
  hasError?: boolean;

  /** Error message to display */
  errorMessage?: string;

  /** Whether to auto-focus first input on mount */
  autoFocus?: boolean;

  /** Whether inputs are secure (hide digits) */
  secureTextEntry?: boolean;

  /** Placeholder character for empty inputs */
  placeholder?: string;

  /** Test ID for testing */
  testID?: string;

  /** Accessibility label */
  accessibilityLabel?: string;
}

/**
 * Individual OTP Input Field Props
 */
export interface OTPInputFieldProps {
  /** Current digit value */
  value: string;

  /** Field index */
  index: number;

  /** Whether this field is focused */
  isFocused: boolean;

  /** Whether to show error state */
  hasError: boolean;

  /** Whether the field is disabled */
  disabled: boolean;

  /** Whether to secure text entry */
  secureTextEntry: boolean;

  /** Placeholder character */
  placeholder: string;

  /** Callback when value changes */
  onChangeText: (text: string, index: number) => void;

  /** Callback when key is pressed */
  onKeyPress: (key: string, index: number) => void;

  /** Callback when field is focused */
  onFocus: (index: number) => void;

  /** Callback when field loses focus */
  onBlur: (index: number) => void;

  /** Ref for the TextInput */
  inputRef: React.RefObject<any>;

  /** Test ID */
  testID?: string;
}

/**
 * OTP Input State
 */
export interface OTPInputState {
  /** Array of digit values */
  digits: string[];

  /** Currently focused field index */
  focusedIndex: number;

  /** Whether any field is focused */
  isFocused: boolean;
}
