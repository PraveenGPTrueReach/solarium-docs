import React from 'react';
import {View, Text, StyleSheet} from 'react-native';
import MaterialIcons from 'react-native-vector-icons/MaterialIcons';
import type {CartItem} from '../../../store/servicesCartSlice';
import {useAppTheme} from '../../../theme/ThemeProvider';

export interface CartItemRowProps {
  /** Cart item to display */
  cartItem: CartItem;
  /** Whether to show quantity controls (disabled for checkout) */
  readonly?: boolean;
  /** Test ID for testing */
  testID?: string;
}

/**
 * CartItemRow Component
 */
const CartItemRow: React.FC<CartItemRowProps> = ({
  cartItem,
  readonly = true,
  testID,
}) => {
  const theme = useAppTheme();
  const {service, qty} = cartItem;

  return (
    <View
      style={[styles.container, {backgroundColor: theme.colors.surface}]}
      testID={testID}
      accessible={true}
      accessibilityRole="text"
      accessibilityLabel={`${service.name}, ${service.type} service, quantity ${qty}. ${service.description}`}
    >
      {/* Service Header */}
      <View style={styles.header}>
        <MaterialIcons
          name="solar-power"
          size={20}
          color={theme.colors.primary}
          style={styles.icon}
        />
        <View style={styles.serviceInfo}>
          <Text
            style={[styles.serviceName, {color: theme.colors.onSurface}]}
            numberOfLines={1}
          >
            {service.name}
          </Text>
          <Text
            style={[styles.serviceType, {color: theme.colors.onSurfaceVariant}]}
            numberOfLines={1}
          >
            {service.type}
          </Text>
        </View>
        <View style={styles.quantityContainer}>
          <Text
            style={[
              styles.quantityLabel,
              {color: theme.colors.onSurfaceVariant},
            ]}
          >
            Qty:
          </Text>
          <Text
            style={[styles.quantityValue, {color: theme.colors.primary}]}
            accessible={true}
            accessibilityRole="text"
            accessibilityLabel={`Quantity: ${qty}`}
          >
            {qty}
          </Text>
        </View>
      </View>

      {/* Service Description */}
      <Text
        style={[styles.description, {color: theme.colors.onSurfaceVariant}]}
        numberOfLines={2}
      >
        {service.description}
      </Text>

      {/* Footer with pricing info */}
      <View style={styles.footer}>
        {service.priceRange && (
          <View style={styles.priceContainer}>
            <MaterialIcons
              name="currency-rupee"
              size={14}
              color={theme.colors.primary}
            />
            <Text style={[styles.priceText, {color: theme.colors.primary}]}>
              Contact for pricing
            </Text>
          </View>
        )}
      </View>
    </View>
  );
};

const styles = StyleSheet.create({
  container: {
    padding: 12,
    marginBottom: 8,
    borderRadius: 8,
    elevation: 1,
    shadowColor: '#000',
    shadowOffset: {width: 0, height: 1},
    shadowOpacity: 0.1,
    shadowRadius: 2,
  },
  header: {
    flexDirection: 'row',
    alignItems: 'center',
    marginBottom: 6,
  },
  icon: {
    marginRight: 8,
  },
  serviceInfo: {
    flex: 1,
  },
  serviceName: {
    fontSize: 14,
    fontWeight: '600',
    marginBottom: 2,
  },
  serviceType: {
    fontSize: 12,
    fontWeight: '400',
  },
  quantityContainer: {
    flexDirection: 'row',
    alignItems: 'center',
    backgroundColor: 'rgba(0,0,0,0.05)',
    paddingHorizontal: 8,
    paddingVertical: 4,
    borderRadius: 4,
  },
  quantityLabel: {
    fontSize: 12,
    marginRight: 4,
  },
  quantityValue: {
    fontSize: 12,
    fontWeight: '600',
  },
  description: {
    fontSize: 12,
    lineHeight: 16,
    marginBottom: 6,
  },
  footer: {
    flexDirection: 'row',
    justifyContent: 'flex-end',
  },
  priceContainer: {
    flexDirection: 'row',
    alignItems: 'center',
  },
  priceText: {
    fontSize: 12,
    fontWeight: '500',
    marginLeft: 2,
  },
});

export default CartItemRow;
