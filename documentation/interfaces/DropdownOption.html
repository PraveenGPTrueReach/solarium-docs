import React, {useState, useCallback} from 'react';
import {View, StyleSheet, FlatList} from 'react-native';
import {
  TextInput,
  Menu,
  Divider,
  Text,
  TouchableRipple,
} from 'react-native-paper';
import MaterialIcons from 'react-native-vector-icons/MaterialIcons';
import {useAppTheme} from '../../../theme/ThemeProvider';

export interface DropdownOption {
  label: string;
  value: string;
  sublabel?: string;
}

export interface DropdownProps {
  label: string;
  placeholder?: string;
  value?: string;
  options: DropdownOption[];
  onSelect: (value: string) => void;
  error?: string;
  disabled?: boolean;
  loading?: boolean;
  testID?: string;
}

/**
 * Dropdown Component
 */
const Dropdown: React.FC<DropdownProps> = ({
  label,
  placeholder = 'Select an option...',
  value,
  options,
  onSelect,
  error,
  disabled = false,
  loading = false,
  testID,
}) => {
  const theme = useAppTheme();
  const [visible, setVisible] = useState(false);

  const selectedOption = options.find(option => option.value === value);

  const openMenu = useCallback(() => {
    if (!disabled && !loading) {
      setVisible(true);
    }
  }, [disabled, loading]);

  const closeMenu = useCallback(() => {
    setVisible(false);
  }, []);

  const handleSelect = useCallback(
    (selectedValue: string) => {
      onSelect(selectedValue);
      closeMenu();
    },
    [onSelect, closeMenu]
  );

  const renderOption = useCallback(
    ({item}: {item: DropdownOption}) => (
      <TouchableRipple
        onPress={() => handleSelect(item.value)}
        style={styles.menuItem}
        testID={`${testID}-option-${item.value}`}
      >
        <View>
          <Text variant="bodyMedium" style={{color: theme.colors.onSurface}}>
            {item.label}
          </Text>
          {item.sublabel && (
            <Text
              variant="bodySmall"
              style={{color: theme.colors.onSurfaceVariant}}
            >
              {item.sublabel}
            </Text>
          )}
        </View>
      </TouchableRipple>
    ),
    [handleSelect, theme, testID]
  );

  const keyExtractor = useCallback((item: DropdownOption) => item.value, []);

  return (
    <View style={styles.container}>
      <Menu
        visible={visible}
        onDismiss={closeMenu}
        contentStyle={[
          styles.menuContent,
          {backgroundColor: theme.colors.surface},
        ]}
        anchor={
          <TextInput
            label={label}
            value={selectedOption?.label || ''}
            placeholder={placeholder}
            onChangeText={() => {}} // Controlled by dropdown
            onTouchStart={openMenu}
            editable={false}
            error={!!error}
            disabled={disabled}
            right={
              <TextInput.Icon
                icon={({color, size}) => (
                  <MaterialIcons
                    name={visible ? 'keyboard-arrow-up' : 'keyboard-arrow-down'}
                    size={size}
                    color={color}
                  />
                )}
                onPress={openMenu}
                disabled={disabled || loading}
              />
            }
            style={[
              styles.textInput,
              disabled && {backgroundColor: theme.colors.surfaceDisabled},
            ]}
            testID={testID}
            accessibilityLabel={label}
            accessibilityHint="Tap to open dropdown menu"
          />
        }
        anchorPosition="bottom"
      >
        {loading ? (
          <View style={styles.loadingContainer}>
            <Text style={{color: theme.colors.onSurfaceVariant}}>
              Loading options...
            </Text>
          </View>
        ) : options.length === 0 ? (
          <View style={styles.emptyContainer}>
            <Text style={{color: theme.colors.onSurfaceVariant}}>
              No options available
            </Text>
          </View>
        ) : (
          <FlatList
            data={options}
            renderItem={renderOption}
            keyExtractor={keyExtractor}
            ItemSeparatorComponent={() => <Divider />}
            style={styles.optionsList}
            nestedScrollEnabled={true}
          />
        )}
      </Menu>

      {error && (
        <Text
          variant="bodySmall"
          style={[styles.errorText, {color: theme.colors.error}]}
        >
          {error}
        </Text>
      )}
    </View>
  );
};

const styles = StyleSheet.create({
  container: {
    marginBottom: 16,
  },
  textInput: {
    backgroundColor: 'transparent',
  },
  menuContent: {
    borderRadius: 8,
    elevation: 8,
    maxHeight: 200,
  },
  optionsList: {
    maxHeight: 180,
  },
  menuItem: {
    padding: 12,
    minHeight: 48,
    justifyContent: 'center',
  },
  loadingContainer: {
    padding: 16,
    alignItems: 'center',
  },
  emptyContainer: {
    padding: 16,
    alignItems: 'center',
  },
  errorText: {
    marginTop: 4,
    marginLeft: 16,
  },
});

export default Dropdown;
