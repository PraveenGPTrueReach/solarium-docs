export class ServiceFetchError extends Error {
  public readonly code: string;
  public readonly operation: string;
  public readonly serviceId?: string;
  public readonly httpStatus?: number;
  public readonly originalError?: Error;

  constructor(
    message: string,
    operation: string,
    code: string = 'SERVICE_ERROR',
    options?: {
      serviceId?: string;
      httpStatus?: number;
      originalError?: Error;
    }
  ) {
    super(message);
    this.name = 'ServiceFetchError';
    this.code = code;
    this.operation = operation;
    this.serviceId = options?.serviceId;
    this.httpStatus = options?.httpStatus;
    this.originalError = options?.originalError;

    // Maintains proper stack trace for where our error was thrown (V8 only)
    if (Error.captureStackTrace) {
      Error.captureStackTrace(this, ServiceFetchError);
    }
  }

  /**
   * Create error for service list fetch operations
   */
  static forListFetch(
    originalError?: Error,
    httpStatus?: number
  ): ServiceFetchError {
    return new ServiceFetchError(
      'Failed to fetch services list',
      'LIST_FETCH',
      'LIST_FETCH_ERROR',
      {originalError, httpStatus}
    );
  }

  /**
   * Create error for individual service fetch operations
   */
  static forServiceFetch(
    serviceId: string,
    originalError?: Error,
    httpStatus?: number
  ): ServiceFetchError {
    return new ServiceFetchError(
      `Failed to fetch service: ${serviceId}`,
      'SERVICE_FETCH',
      'SERVICE_FETCH_ERROR',
      {serviceId, originalError, httpStatus}
    );
  }

  /**
   * Convert to plain object for logging
   */
  toJSON() {
    return {
      name: this.name,
      message: this.message,
      code: this.code,
      operation: this.operation,
      serviceId: this.serviceId,
      httpStatus: this.httpStatus,
      stack: this.stack,
      originalError: this.originalError?.message,
    };
  }
}
