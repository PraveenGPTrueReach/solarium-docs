/**
 * Upload error with step information
 */
export class KycUploadError extends Error {
  public readonly step: string;
  public readonly originalError: Error;

  constructor(step: string, originalError: Error) {
    super(`KYC upload failed at ${step}: ${originalError.message}`);
    this.name = 'KycUploadError';
    this.step = step;
    this.originalError = originalError;
  }
}

/**
 * Mutex for SAS token refresh to prevent concurrent requests
 */
class SasTokenMutex {
  private readonly locks = new Map<string, Promise<SasInfo>>();

  async executeWithLock(
    key: string,
    operation: () => Promise<SasInfo>
  ): Promise<SasInfo> {
    // If there's already a pending operation for this key, wait for it
    if (this.locks.has(key)) {
      return this.locks.get(key)!;
    }

    // Create and store the operation promise
    const promise = operation();
    this.locks.set(key, promise);

    try {
      const result = await promise;
      return result;
    } finally {
      // Always clean up the lock
      this.locks.delete(key);
    }
  }
}

/**
 * Extract blob name from SAS URL
 * Format: https://storage.blob.core.windows.net/container/path/filename.ext?sas=token
 */
function extractBlobName(sasUrl: string): string {
  try {
    const url = new URL(sasUrl);
    const pathParts = url.pathname.split('/');

    // Remove empty first element and container name
    const pathWithoutContainer = pathParts.slice(2);

    // Join remaining parts to get blob name
    return pathWithoutContainer.join('/');
  } catch (error) {
    console.warn(
      '[KycUploader] Failed to extract blob name from SAS URL:',
      error
    );
    // Fallback: use timestamp-based name
    return `unknown-${Date.now()}`;
  }
}

/**
 * Extract base URL (without query parameters) from SAS URL
 */
function extractBaseUrl(sasUrl: string): string {
  try {
    const url = new URL(sasUrl);
    return `${url.origin}${url.pathname}`;
  } catch (error) {
    console.warn(
      '[KycUploader] Failed to extract base URL from SAS URL:',
      error
    );
    return sasUrl.split('?')[0];
  }
}

/**
 * Get current user ID from auth state
 */
function getCurrentUserId(store?: {getState: () => RootState}): string {
  if (!store) {
    throw new Error('Store is required to get current user ID');
  }

  const state = store.getState();

  // Get user from new auth system
  const newAuth = selectNewAuth(state);
  const user = newAuth?.user;

  if (!user || !user.id) {
    throw new Error('User not authenticated or missing user ID');
  }

  return user.id;
}

/**
 * Check if SAS token needs refresh based on remaining validity and estimated upload time
 */
function needsSasTokenRefresh(
  sasInfo: SasInfo,
  file: RNFile,
  bufferTimeSeconds: number = 60
): boolean {
  // Calculate estimated upload time (rough estimate: 1MB/s)
  const estimatedUploadTimeSeconds = estimateUploadTime(file.size);

  // Check if remaining validity is less than estimated upload time + buffer
  const requiredTimeSeconds = estimatedUploadTimeSeconds + bufferTimeSeconds;

  const needsRefresh = sasInfo.expiresInSec < requiredTimeSeconds;

  console.log('[KycUploader] SAS token validity check:', {
    expiresInSec: sasInfo.expiresInSec,
    estimatedUploadTimeSeconds,
    bufferTimeSeconds,
    requiredTimeSeconds,
    needsRefresh,
  });

  return needsRefresh;
}

/**
 * Upload KYC document with complete orchestration
 *
 * Workflow:
 * 1. Validate document (MIME, size, count)
 * 2. Compress if image > 2MB
 * 3. Get SAS token with smart expiry checking
 * 4. Upload to Azure Blob with progress tracking
 * 5. Invalidate cache tags
 * 6. Return blob information
 *
 * @param params - Upload parameters
 * @returns Promise resolving to blob information
 */
export async function uploadKycDocument(
  params: KycDocUploadParams
): Promise<UploadResult> {
  const {
    leadId,
    customerId,
    docType,
    file,
    onProgress,
    store,
    onInvalidate,
    abortSignal,
  } = params;

  console.log(`[KycUploader] Starting KYC document upload for ${file.name}`);

  let processedFile = file;
  let sasInfo: SasInfo;

  // Step 0: Validate customer ID
  try {
    console.log('[KycUploader] Step 3: Validating customer ID...');
    if (!customerId) {
      throw new Error('Customer ID is required');
    }
    console.log(`[KycUploader] Using customer ID: ${customerId}`);
  } catch (error) {
    console.error('[KycUploader] Customer ID validation failed:', error);
    throw new KycUploadError('authentication', error as Error);
  }

  // Step 1: Validate document
  try {
    console.log('[KycUploader] Step 1: Validating document...');
    await validateKycDocument(file, customerId, leadId);
    console.log('[KycUploader] Document validation passed');
  } catch (error) {
    console.error('[KycUploader] Document validation failed:', error);
    throw new KycUploadError('validation', error as Error);
  }

  // Step 2: Compress if needed
  try {
    console.log('[KycUploader] Step 2: Checking compression needs...');

    const isImage = isCompressibleImage(file);
    const sizeMB = file.size / (1024 * 1024);

    if (isImage && sizeMB > 2) {
      console.log(
        `[KycUploader] Compressing ${file.name} (${sizeMB.toFixed(2)} MB)`
      );
      // Update progress for compression
      if (onProgress) {
        onProgress(10); // 10% for compression start
      }

      const compressionResult = await compressImage(file);
      processedFile = compressionResult;

      console.log(
        `[KycUploader] Compression completed: ${file.size} â†’ ${compressionResult.size} bytes`
      );
      if (onProgress) {
        onProgress(20); // 20% for compression complete
      }
    } else {
      console.log('[KycUploader] No compression needed');
      if (onProgress) {
        onProgress(20); // Skip compression, move to 20%
      }
    }
  } catch (error) {
    console.error('[KycUploader] Compression failed:', error);
    throw new KycUploadError('compression', error as Error);
  }

  // Step 3: Get SAS token with smart expiry checking
  try {
    console.log('[KycUploader] Step 3: Requesting SAS token...');

    if (onProgress) {
      onProgress(30); // 30% for SAS token request
    }

    let requestKycSasParams: RequestSasTokenRequest;
    if (leadId) {
      requestKycSasParams = {
        customerId,
        docType,
        leadId,
      };
    } else {
      requestKycSasParams = {
        customerId,
        docType,
        leadId: 'LEAD-1014',
      };
    }
    sasInfo = await requestKycSasToken(requestKycSasParams, store);

    console.log(
      `[KycUploader] SAS token acquired, expires in ${sasInfo.expiresInSec}s`
    );

    // Check if we need to refresh token immediately
    if (needsSasTokenRefresh(sasInfo, processedFile)) {
      console.log(
        '[KycUploader] SAS token needs immediate refresh due to insufficient validity'
      );
      sasInfo = await requestKycSasToken(requestKycSasParams, store);
      console.log(
        `[KycUploader] Fresh SAS token acquired, expires in ${sasInfo.expiresInSec}s`
      );
    }

    if (onProgress) {
      onProgress(40); // 40% for SAS token ready
    }
  } catch (error) {
    console.error('[KycUploader] SAS token acquisition failed:', error);
    throw new KycUploadError('sas_token', error as Error);
  }

  // Step 4: Upload to Azure Blob
  try {
    console.log('[KycUploader] Step 4: Uploading to Azure Blob...');

    const getFreshSas = async (): Promise<SasInfo> => {
      console.log('[KycUploader] Refreshing SAS token during upload...');

      let requestKycSasParams: RequestSasTokenRequest;
      if (leadId) {
        requestKycSasParams = {
          customerId,
          docType,
          leadId,
        };
      } else {
        requestKycSasParams = {
          customerId,
          docType,
          leadId: 'LEAD-1013',
        };
      }
      return sasTokenMutex.executeWithLock(
        `${leadId}-${docType}-${customerId}`,
        async () => {
          const freshSas = await requestKycSasToken(requestKycSasParams, store);
          console.log(
            `[KycUploader] Fresh SAS token acquired during upload, expires in ${freshSas.expiresInSec}s`
          );
          return freshSas;
        }
      );
    };

    await uploadFileToSas({
      // This function now expects AzureBlobUploadParams
      sasUrl: sasInfo.sasUrl,
      file: processedFile,
      expiresInSec: sasInfo.expiresInSec,
      onProgress: progress => {
        // Map upload progress to 40-90% range
        const mappedProgress = 40 + progress.percentage * 0.5;
        if (onProgress) {
          onProgress(Math.round(mappedProgress));
        }
      },
      getFreshSas,
      abortSignal, // New parameter abortSignal for cancellation
    });

    console.log('[KycUploader] Upload to Azure Blob completed');

    if (onProgress) {
      onProgress(90); // 90% for upload complete
    }
  } catch (error) {
    console.error('[KycUploader] Upload to Azure Blob failed:', error);
    throw new KycUploadError('upload', error as Error);
  }

  // Step 5: Invalidate cache tags
  try {
    console.log('[KycUploader] Step 5: Invalidating cache tags...');

    if (store) {
      // Prefer store-based invalidation
      console.log('[KycUploader] Using store-based cache invalidation');
      store.dispatch(
        kycApi.util.invalidateTags([
          {type: 'KycDoc', id: leadId},
          {type: 'KycDoc', id: customerId},
          {type: 'KycDoc', id: 'LIST'},
        ])
      );
    } else if (onInvalidate) {
      // Fallback to callback-based invalidation
      console.log('[KycUploader] Using callback-based cache invalidation');
      onInvalidate(leadId);
    } else {
      console.warn(
        '[KycUploader] No invalidation method provided - cache may be stale'
      );
    }

    if (onProgress) {
      onProgress(95); // 95% for cache invalidation
    }
  } catch (error) {
    console.error('[KycUploader] Cache invalidation failed:', error);
    // Don't throw here - upload succeeded, cache invalidation is nice-to-have
  }

  // Step 6: Return result
  try {
    console.log('[KycUploader] Step 6: Preparing result...');

    const blobName = extractBlobName(sasInfo.sasUrl);
    const url = extractBaseUrl(sasInfo.sasUrl);

    const result: UploadResult = {
      blobName,
      url,
    };

    console.log(
      '[KycUploader] KYC document upload completed successfully:',
      result
    );
    if (onProgress) {
      onProgress(100); // 100% complete
    }

    return result;
  } catch (error) {
    console.error('[KycUploader] Failed to prepare result:', error);
    throw new KycUploadError('result_preparation', error as Error);
  }
}

/**
 * Utility function to check if a file needs compression
 */
export function shouldCompressFile(file: RNFile): boolean {
  const isImage = isCompressibleImage(file);
  const sizeMB = file.size / (1024 * 1024);
  return isImage && sizeMB > 2;
}

/**
 * Utility function to estimate total upload time including all steps
 */
export function estimateKycUploadTime(file: RNFile): number {
  let totalSeconds = 0;

  // Validation: ~1 second (includes API call for doc count)
  totalSeconds += 1;

  // Compression: ~2 seconds for 10MB image
  if (shouldCompressFile(file)) {
    totalSeconds += 2;
  }

  // SAS token: ~0.5 seconds
  totalSeconds += 0.5;

  // Upload: file size dependent
  totalSeconds += estimateUploadTime(file.size);

  // Cache invalidation: ~0.1 seconds
  totalSeconds += 0.1;

  return Math.ceil(totalSeconds);
}
