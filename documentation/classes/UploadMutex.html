import type {
  SasInfo,
  AzureBlobUploadParams,
  UploadProgress,
  UploadError,
  UploadErrorType,
} from '../types/AzureBlob';

/**
 * Upload configuration
 */
const UPLOAD_CONFIG = {
  // Timeout based on environment (in milliseconds)
  TIMEOUT_MS: __DEV__ ? 10000 : 8000, // 10s dev/staging, 8s prod

  // Progress update interval
  PROGRESS_INTERVAL_MS: 100,

  // Minimum progress ticks for 10MB file
  MIN_PROGRESS_TICKS: 5,

  // Maximum retry attempts
  MAX_RETRY_ATTEMPTS: 1,

  // Buffer time for token expiry checks (30 seconds)
  EXPIRY_BUFFER_MS: 30 * 1000,
} as const;

/**
 * Mutex to prevent concurrent getFreshSas calls
 */
class UploadMutex {
  private readonly locks = new Map<string, Promise<SasInfo>>();

  async executeWithLock<T>(
    key: string,
    operation: () => Promise<T>
  ): Promise<T> {
    // If there's already a pending operation for this key, wait for it
    if (this.locks.has(key)) {
      return this.locks.get(key) as Promise<T>;
    }

    // Create and store the operation promise
    const promise = operation();
    this.locks.set(key, promise as Promise<SasInfo>);

    try {
      const result = await promise;
      return result;
    } finally {
      // Always clean up the lock
      this.locks.delete(key);
    }
  }
}

// Global mutex instance
const uploadMutex = new UploadMutex();

/**
 * Create upload error with specific type
 */
function createUploadError(
  message: string,
  type: UploadErrorType,
  statusCode?: number,
  retryable: boolean = false
): UploadError {
  const error = new Error(message) as UploadError;
  error.type = type;
  error.statusCode = statusCode;
  error.retryable = retryable;
  return error;
}

/**
 * Calculate progress with minimum tick guarantee
 */
function calculateProgress(loaded: number, total: number): UploadProgress {
  const percentage = total > 0 ? Math.round((loaded / total) * 100) : 0;

  return {
    loaded,
    total,
    percentage: Math.min(percentage, 100),
  };
}

/**
 * Check if upload has expired based on start time and expiry
 */
function isUploadExpired(startTime: number, expiresInSec: number): boolean {
  const elapsedSeconds = (Date.now() - startTime) / 1000;
  return elapsedSeconds >= expiresInSec - UPLOAD_CONFIG.EXPIRY_BUFFER_MS / 1000;
}

/**
 * Perform the actual upload using XMLHttpRequest
 */
function performUpload(
  sasUrl: string,
  file: any, // RNFile
  onProgress?: (progress: UploadProgress) => void,
  abortSignal?: AbortSignal
): Promise<void> {
  return new Promise((resolve, reject) => {
    const xhr = new XMLHttpRequest();
    let lastProgressTime = 0;

    // Setup progress tracking
    const updateProgress = (loaded: number, total: number) => {
      const now = Date.now();

      // Ensure minimum progress interval (100ms)
      if (now - lastProgressTime >= UPLOAD_CONFIG.PROGRESS_INTERVAL_MS) {
        lastProgressTime = now;

        if (onProgress) {
          const progress = calculateProgress(loaded, total);
          onProgress(progress);

          console.log(
            `[AzureBlobUploader] Progress: ${progress.percentage}% (${loaded}/${total} bytes)`
          );
        }
      }
    };

    // Handle upload progress
    xhr.upload.addEventListener('progress', event => {
      if (event.lengthComputable) {
        updateProgress(event.loaded, event.total);
      }
    });

    // Handle completion
    xhr.addEventListener('load', () => {
      if (xhr.status === 200 || xhr.status === 201) {
        console.log(
          `[AzureBlobUploader] Upload completed successfully: ${xhr.status}`
        );
        // Ensure final progress is 100%
        if (onProgress) {
          onProgress(calculateProgress(100, 100));
        }

        resolve();
      } else {
        console.error(
          `[AzureBlobUploader] Upload failed with status: ${xhr.status}`
        );
        const error = createUploadError(
          `Upload failed with status ${xhr.status}`,
          xhr.status === 403 ? 'EXPIRED_TOKEN' : 'INVALID_RESPONSE',
          xhr.status,
          xhr.status === 403 // 403 is retryable
        );

        reject(error);
      }
    });

    // Handle network errors
    xhr.addEventListener('error', () => {
      console.error('[AzureBlobUploader] Network error during upload');
      const error = createUploadError(
        'Network error during upload',
        'NETWORK_ERROR',
        undefined,
        true // Network errors are retryable
      );
      reject(error);
    });

    // Handle timeout
    xhr.addEventListener('timeout', () => {
      console.error('[AzureBlobUploader] Upload timeout');
      const error = createUploadError(
        'Upload timeout',
        'TIMEOUT_ERROR',
        undefined,
        true // Timeouts are retryable
      );
      reject(error);
    });

    // Handle abort
    xhr.addEventListener('abort', () => {
      console.log('[AzureBlobUploader] Upload cancelled');
      const error = createUploadError(
        'Upload cancelled',
        'CANCELLED',
        undefined,
        false // Cancelled uploads are not retryable
      );
      reject(error);
    });

    // Setup abort signal
    if (abortSignal) {
      const abortHandler = () => {
        xhr.abort();
      };

      abortSignal.addEventListener('abort', abortHandler);

      // Check if already aborted
      if (abortSignal.aborted) {
        xhr.abort();
        return;
      }
    }

    // Configure and start upload
    xhr.open('PUT', sasUrl);
    xhr.timeout = UPLOAD_CONFIG.TIMEOUT_MS;

    // Set content type based on file type
    if (file.type) {
      xhr.setRequestHeader('Content-Type', file.type);
    }

    // Additional Azure Blob headers
    xhr.setRequestHeader('x-ms-blob-type', 'BlockBlob');

    console.log(
      `[AzureBlobUploader] Starting upload: ${file.name} (${file.size} bytes)`
    );
    // Start upload with file blob
    if (file.uri && file.uri.startsWith('file://')) {
      // For React Native file URIs, we need to create a blob-like object
      // This is a simplified approach - in a real app, you'd use libraries like react-native-fs
      xhr.send({
        uri: file.uri,
        type: file.type,
        name: file.name,
      } as any);
    } else {
      // For web blobs or direct file objects
      xhr.send(file as any);
    }
  });
}

/**
 * Upload file to Azure Blob Storage using SAS URL
 *
 * @param params - Upload parameters
 * @returns Promise that resolves when upload completes
 *
 * Features:
 * - Progress tracking with minimum 100ms intervals
 * - Automatic retry on 403 (expired token) or timeout
 * - Global upload expiry check
 * - Cancellation support via AbortSignal
 * - Mutex to prevent concurrent token refresh
 */
export async function uploadFileToSas(
  params: AzureBlobUploadParams
): Promise<void> {
  const {sasUrl, file, expiresInSec, onProgress, getFreshSas, abortSignal} =
    params;

  const startTime = Date.now();
  const uploadKey = `${file.name}-${file.size}-${startTime}`;
  let currentSasUrl = sasUrl;
  let attempts = 0;

  console.log(
    `[AzureBlobUploader] Starting upload for ${file.name} with ${expiresInSec}s expiry`
  );

  // Validation
  if (!sasUrl || !file) {
    throw createUploadError(
      'Invalid upload parameters: sasUrl and file are required',
      'INVALID_RESPONSE'
    );
  }

  if (file.size <= 0) {
    throw createUploadError(
      'Invalid file: file size must be greater than 0',
      'INVALID_RESPONSE'
    );
  }

  while (attempts <= UPLOAD_CONFIG.MAX_RETRY_ATTEMPTS) {
    // Check if upload has expired
    if (isUploadExpired(startTime, expiresInSec)) {
      console.error('[AzureBlobUploader] Upload expired due to time limit');
      throw createUploadError(
        `Upload expired: total time exceeded ${expiresInSec} seconds`,
        'UPLOAD_EXPIRED'
      );
    }

    // Check if aborted
    if (abortSignal?.aborted) {
      throw createUploadError('Upload cancelled by user', 'CANCELLED');
    }

    try {
      console.log(`[AzureBlobUploader] Upload attempt ${attempts + 1}`);

      await performUpload(currentSasUrl, file, onProgress, abortSignal);

      console.log(
        `[AzureBlobUploader] Upload completed successfully after ${
          attempts + 1
        } attempt(s)`
      );
      return;
    } catch (error) {
      const uploadError = error as UploadError;
      attempts++;

      console.error(
        `[AzureBlobUploader] Upload attempt ${attempts} failed:`,
        uploadError.message
      );

      // Check if we should retry
      const shouldRetry =
        attempts <= UPLOAD_CONFIG.MAX_RETRY_ATTEMPTS &&
        uploadError.retryable &&
        getFreshSas &&
        !abortSignal?.aborted;

      if (!shouldRetry) {
        console.error('[AzureBlobUploader] No more retries, failing upload');
        throw uploadError;
      }

      // Only retry on 403 (expired token) or timeout
      if (
        uploadError.type === 'EXPIRED_TOKEN' ||
        uploadError.type === 'TIMEOUT_ERROR'
      ) {
        console.log('[AzureBlobUploader] Attempting to refresh SAS token...');

        try {
          // Use mutex to prevent concurrent getFreshSas calls
          const freshSasInfo = await uploadMutex.executeWithLock(
            `refresh-${uploadKey}`,
            getFreshSas
          );

          currentSasUrl = freshSasInfo.sasUrl;
          console.log(
            '[AzureBlobUploader] SAS token refreshed, retrying upload...'
          );
          // Reset progress to 0 for retry
          if (onProgress) {
            onProgress(calculateProgress(0, file.size));
          }
        } catch (refreshError) {
          console.error(
            '[AzureBlobUploader] Failed to refresh SAS token:',
            refreshError
          );
          throw createUploadError(
            `Failed to refresh SAS token: ${(refreshError as Error).message}`,
            'EXPIRED_TOKEN'
          );
        }
      } else {
        // For non-retryable errors, fail immediately
        throw uploadError;
      }
    }
  }

  // If we get here, all retries failed
  throw createUploadError(
    `Upload failed after ${UPLOAD_CONFIG.MAX_RETRY_ATTEMPTS + 1} attempts`,
    'NETWORK_ERROR'
  );
}

/**
 * Create an AbortController for upload cancellation
 *
 * @returns AbortController instance
 */
export function createUploadAbortController(): AbortController {
  return new AbortController();
}

/**
 * Get upload configuration
 */
export function getUploadConfig() {
  return UPLOAD_CONFIG;
}

/**
 * Estimate upload time based on file size (rough estimate)
 *
 * @param fileSizeBytes - File size in bytes
 * @param connectionSpeedBps - Connection speed in bytes per second (default: 1MB/s)
 * @returns Estimated upload time in seconds
 */
export function estimateUploadTime(
  fileSizeBytes: number,
  connectionSpeedBps: number = 1024 * 1024 // 1MB/s default
): number {
  return Math.ceil(fileSizeBytes / connectionSpeedBps);
}
